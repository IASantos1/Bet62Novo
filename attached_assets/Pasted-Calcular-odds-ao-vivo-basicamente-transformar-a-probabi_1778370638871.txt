⚙️ 5. SCRIPT REAL (LIVE ODDS)
⚽ Futebol
function calculate1x2(state) {

  let homeProb = 0.45;
  let drawProb = 0.30;
  let awayProb = 0.25;

  // vantagem no placar
  if (state.home_goals > state.away_goals) {
    homeProb += 0.20;
    awayProb -= 0.10;
  }

  // tempo do jogo
  if (state.minute > 70) {
    homeProb += 0.10;
    drawProb -= 0.05;
  }

  // cartão vermelho
  if (state.red_cards_away > 0) {
    homeProb += 0.08;
  }

  // normalização
  const total =
    homeProb + drawProb + awayProb;

  homeProb /= total;
  drawProb /= total;
  awayProb /= total;

  // odds
  return {

    home:
      ((1 / homeProb) * 0.94).toFixed(2),

    draw:
      ((1 / drawProb) * 0.94).toFixed(2),

    away:
      ((1 / awayProb) * 0.94).toFixed(2)
  };
}

function calculateOver25(state) {

  const totalXG =
    state.home_xg + state.away_xg;

  let prob = totalXG / 3;

  prob = Math.min(0.95, prob);

  return {

    over25:
      ((1 / prob) * 0.94).toFixed(2),

    under25:
      ((1 / (1 - prob)) * 0.94).toFixed(2)
  };
}
