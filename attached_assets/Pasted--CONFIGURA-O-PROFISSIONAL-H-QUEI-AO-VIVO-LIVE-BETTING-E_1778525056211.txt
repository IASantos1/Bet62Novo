function onPeriodEnd(match) {

  closeMarkets(
    match.id,
    `P${match.period}`
  );

  match.period += 1;

  openMarkets(
    match.id,
    `P${match.period}`
  );

  recalculateAllOdds(match);
}

⚙️ SCRIPT
function updateGoalLines(match) {

  const totalGoals =
    match.score.home +
    match.score.away;

  match.markets.totals.lines =
    match.markets.totals.lines
      .filter(line => line.value > totalGoals);

  const nextLines = [

    totalGoals + 0.5,
    totalGoals + 1.5

  ];

  nextLines.forEach(line => {

    addLineIfMissing(match, line);

  });
}
