⚙️ 3. FETCH STATPAL
async function syncPrematch() {

  const response =
    await axios.get(
      "https://api.statpal.io/prematch"
    );

  const events = response.data;

  events.forEach(event => {

    prematchCache[event.id] = event;

  });

  cleanupEvents(events);
}
⚡ 4. REMOVER JOGOS INICIADOS

Você perguntou exatamente isso.

Fluxo:
MATCH STARTS
    ↓
STATUS = LIVE
    ↓
REMOVE PREMATCH
SCRIPT
function cleanupEvents(events) {

  Object.keys(prematchCache)
    .forEach(id => {

      const event =
        prematchCache[id];

      if (
        event.status === "live" ||
        event.status === "ended"
      ) {

        delete prematchCache[id];
      }
    });
}

Fluxo correto:
NOVOS → adiciona

ALTERADOS → atualiza
SCRIPT
function updateEvents(newEvents) {

  newEvents.forEach(event => {

    const exists =
      currentEvents[event.id];

    if (!exists) {

      addCard(event);

    } else {

      updateCard(event);

    }

  });

  removeMissingEvents(newEvents);
}
⚡ 8. REMOVER EVENTOS DO FRONT
function removeMissingEvents(events) {

  const ids =
    events.map(e => e.id);

  Object.keys(currentEvents)
    .forEach(id => {

      if (!ids.includes(id)) {

        removeCard(id);

      }

    });
}
