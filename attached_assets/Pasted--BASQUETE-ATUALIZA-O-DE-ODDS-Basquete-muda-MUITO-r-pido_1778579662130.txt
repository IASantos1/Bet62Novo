⚡ PROCESSAMENTO
function processMarkets(event) {

  const config =
    SPORTS_CONFIG[event.sport];

  if (
    config.suspendOn.includes(
      event.lastEvent
    )
  ) {

    suspendMarkets(event.id);
  }

  updateOdds(event);

  updateLines(event);

  removeFinishedMarkets(event);
}
❌ REMOVER MERCADOS ENCERRADOS
function removeFinishedMarkets(event) {

  event.markets =
    event.markets.filter(m => {

      return !m.finished;

    });
}
⚡ GERAR NOVAS LINHAS
function updateLines(event) {

  if (
    event.totalGoals >= 2
  ) {

    addLine("3.5");

    addLine("4.5");
  }
}
