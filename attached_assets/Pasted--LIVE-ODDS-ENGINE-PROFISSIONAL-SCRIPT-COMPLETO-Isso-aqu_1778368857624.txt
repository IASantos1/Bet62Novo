export function shouldSuspend(event) {

  const critical = [
    "goal",
    "var",
    "red_card",
    "penalty",
    "touchdown"
  ];

  return critical.includes(event.type);
}
⚡ 5. MARKET LOCK
export function suspendMarkets(matchId) {

  if (!matchCache[matchId]) return;

  matchCache[matchId].markets.forEach(m => {

    m.status = "suspended";

  });

  return true;
}
🧠 6. ODDS ENGINE
export function calculateOdds(state) {

  const totalGoals =
    state.homeGoals + state.awayGoals;

  return {

    "1x2": {
      home: totalGoals > 0 ? 1.72 : 2.10,
      draw: 3.80,
      away: 5.20
    },

    totals: {
      over25: totalGoals >= 2 ? 1.55 : 2.20,
      under25: totalGoals >= 2 ? 2.40 : 1.65
    }
  };
}

⚠️ 8. RISK ENGINE
export function detectOddsDrift(oldOdd, newOdd) {

  const diff =
    Math.abs(newOdd - oldOdd);

  return diff > 0.40;
}
🔁 9. MARKET REOPEN
export function reopenMarkets(matchId, odds) {

  matchCache[matchId].markets.forEach(m => {

    m.status = "open";

    if (odds[m.id]) {
      m.odds = odds[m.id];
    }

  });
}

⚡ 14. PROCESSAMENTO DE EVENTOS
async function processEvent(event) {

  const parsed = parseEvent(event);

  if (shouldSuspend(parsed)) {

    suspendMarkets(parsed.matchId);

    pushOdds(parsed.matchId);

  }

  updateMatchState(parsed);

  const newOdds =
    calculateOdds(
      matchCache[parsed.matchId].state
    );

  setTimeout(() => {

    reopenMarkets(
      parsed.matchId,
      newOdds
    );

    pushOdds(parsed.matchId);

  }, 8000);
}
