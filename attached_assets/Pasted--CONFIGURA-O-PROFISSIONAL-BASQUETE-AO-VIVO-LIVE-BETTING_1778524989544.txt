function onBasket(match) {

  suspendMarkets(match.id);

  updateScore(match);

  const odds =
    calculateBasketballOdds(match);

  setTimeout(() => {

    reopenMarkets(match.id, odds);

  }, 1500);
}

⚡ 5. FINAL DE QUARTO

Você perguntou algo equivalente ao tênis:

“quando acabar o período, os mercados somem”

Exatamente.

Fluxo:
QUARTER END
    ↓
CLOSE Q1 MARKETS
    ↓
REMOVE OLD MARKETS
    ↓
OPEN Q2 MARKETS
    ↓
RECALCULATE ODDS
⚙️ SCRIPT
function onQuarterEnd(match) {

  closeMarkets(match.id, `Q${match.quarter}`);

  match.quarter += 1;

  openMarkets(
    match.id,
    `Q${match.quarter}`
  );

  recalculateAllOdds(match);
}

⚙️ SCRIPT
function updateTotalLines(match) {

  const totalPoints =
    match.score.home +
    match.score.away;

  match.markets.totals.lines =
    match.markets.totals.lines
      .filter(line => line.value > totalPoints);

  const nextLines = [

    totalPoints + 5.5,
    totalPoints + 10.5

  ];

  nextLines.forEach(line => {

    addLineIfMissing(match, line);

  });
}
