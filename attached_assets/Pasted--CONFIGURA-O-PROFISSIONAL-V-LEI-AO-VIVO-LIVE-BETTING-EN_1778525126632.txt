function onSetEnd(match) {

  closeMarkets(
    match.id,
    `SET_${match.current_set}`
  );

  match.current_set += 1;

  openMarkets(
    match.id,
    `SET_${match.current_set}`
  );

  recalculateAllOdds(match);
}

function updatePointLines(match) {

  const totalPoints =
    match.points.home +
    match.points.away;

  match.markets.total_points.lines =
    match.markets.total_points.lines
      .filter(line => line.value > totalPoints);

  const nextLines = [

    totalPoints + 2.5,
    totalPoints + 4.5

  ];

  nextLines.forEach(line => {

    addLineIfMissing(match, line);

  });
}
