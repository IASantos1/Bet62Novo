Exatamente.

function onSetFinished(match) {

  closeMarkets(match.id, "set_1");

  match.current_set += 1;

  openMarkets(match.id, `set_${match.current_set}`);

  recalculateAllOdds(match);
}

function updateGameLines(match) {

  const totalGames =
    match.games.player1 +
    match.games.player2;

  match.markets.total_games.lines =
    match.markets.total_games.lines
      .filter(line => line.value > totalGames);

  const nextLines = [

    totalGames + 0.5,
    totalGames + 1.5

  ];

  nextLines.forEach(line => {

    addLineIfMissing(match, line);

  });
}
