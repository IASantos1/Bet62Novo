const loadEnvFileIfExists = (filePath: string) => {
  try {
    if (!fs.existsSync(filePath)) return;
    const content = fs.readFileSync(filePath, "utf8");
    const overrideKeys = new Set([
      "STATPAL_ACCESS_KEY",
      "STATPAL_ACCESS_KEY_2",
      "STATPAL_BASE_URL",
      "STATPAL_SOCCER_DAILY_OFFSET",
      "STATPAL_MAX_REQUESTS_PER_DAY",
      "STATPAL_LIVE_ODDS_MARKET_ID"
    ]);
    for (const line of content.split(/\r?\n/)) {
      const trimmed = line.trim();
      if (!trimmed || trimmed.startsWith("#")) continue;
      const eq = trimmed.indexOf("=");
      if (eq <= 0) continue;
      const key = trimmed
        .slice(0, eq)
        .trim()
        .replace(/^\uFEFF/, "");
      const raw = trimmed.slice(eq + 1).trim();
      const value = raw.startsWith("\"") && raw.endsWith("\"") ? raw.slice(1, -1) : raw;
      if (overrideKeys.has(key)) {
        process.env[key] = value;
      } else if (!process.env[key]) {
        process.env[key] = value;
      }
    }
  } catch {
    return;
  }
};

const round2 = (n: number) => Math.round(n * 100) / 100;
const sum = (arr: number[]) => arr.reduce((a, b) => a + b, 0);
const clamp = (n: number, min: number, max: number) => Math.max(min, Math.min(max, n));
const normalizeOdd = (raw: unknown): number | null => {
  const v = typeof raw === "number" ? raw : Number(String(raw ?? "").trim());
  if (!Number.isFinite(v)) return null;
  if (v >= 100 && Math.abs(v - Math.round(v)) < 1e-9) return v / 100;
  return v;
};

const probsToOdds = (probs: number[], overround: number) => {
  const s = sum(probs);
  const base = s > 0 ? probs.map((p) => p / s) : probs.map(() => 0);
  const q = base.map((p) => p * overround);
  return q.map((x) => round2(clamp(1 / Math.max(1e-9, x), 1.01, 100)));
};

const erf = (x: number) => {
  const s = x < 0 ? -1 : 1;
  const ax = Math.abs(x);
  const t = 1 / (1 + 0.3275911 * ax);
  const y =
    1 -
    (((((1.061405429 * t - 1.453152027) * t + 1.421413741) * t - 0.284496736) * t + 0.254829592) * t) *
      Math.exp(-ax * ax);
  return s * y;
};

const normalCdf = (z: number) => 0.5 * (1 + erf(z / Math.SQRT2));

const poissonPmfUpTo = (lambda: number, maxK: number) => {
  const out = new Array(maxK + 1).fill(0) as number[];
  if (!Number.isFinite(lambda) || lambda < 0) return out;
  const p0 = Math.exp(-lambda);
  out[0] = p0;
  for (let k = 1; k <= maxK; k += 1) out[k] = (out[k - 1] * lambda) / k;
  return out;
};

const poissonCdf = (lambda: number, k: number) => {
  const pmf = poissonPmfUpTo(lambda, Math.max(0, Math.floor(k)));
  return sum(pmf);
};

const prematchSoccerModelFromTeams = (args: { competition: string; home: string; away: string; id: string }) => {
  const homeElo = teamElo(args.home);
  const awayElo = teamElo(args.away);
  const homeAdv = 60;
  const diff = homeElo + homeAdv - awayElo;
  const pHomeVsAway = 1 / (1 + Math.exp(-diff / 260));
  const draw = clamp(0.28 - Math.abs(diff) / 1400, 0.16, 0.3);
  const rem = clamp(1 - draw, 0.01, 0.99);
  const pHome = clamp(rem * pHomeVsAway, 0.01, 0.98);
  const pAway = clamp(1 - draw - pHome, 0.01, 0.98);

  const goalDiff = clamp(diff / 900, -0.6, 0.6);
  const totalGoals = clamp(2.65 + Math.min(0.25, Math.abs(goalDiff) * 0.12), 1.8, 3.8);
  const lambdaHome = clamp(totalGoals / 2 + goalDiff / 2, 0.2, 3.2);
  const lambdaAway = clamp(totalGoals / 2 - goalDiff / 2, 0.2, 3.2);

  return { lambdaHome, lambdaAway, pHomeWin: pHome, pDraw: draw, pAwayWin: pAway };
};

const buildSoccerPrematchMarkets = (args: { competition: string; home: string; away: string; id: string }) => {
  const model = prematchSoccerModelFromTeams(args);
  const [home, draw, away] = probsToOdds([model.pHomeWin, model.pDraw, model.pAwayWin], 1.06);

  const totalGoals = model.lambdaHome + model.lambdaAway;
  const pUnder25 = clamp(poissonCdf(totalGoals, 2), 0.02, 0.98);
  const pOver25 = clamp(1 - pUnder25, 0.02, 0.98);
  const [over, under] = probsToOdds([pOver25, pUnder25], 1.06);

  return {
    oneXTwo: [
      { selection: "HOME" as const, price: home },
      { selection: "DRAW" as const, price: draw },
      { selection: "AWAY" as const, price: away }
    ],
    overUnder25: {
      line: 2.5,
      odds: [
        { selection: "OVER" as const, price: over },
        { selection: "UNDER" as const, price: under }
      ]
    }
  };
};

const computePrematchSoccerModel = (seedKey: string) => {
  const parts = seedKey.split(":");
  const home = parts[0] ?? "Home";
  const away = parts[1] ?? "Away";
  const competition = parts[2] ?? "Futebol";
  const id = parts[3] ?? seedKey;
  const base = prematchSoccerModelFromTeams({ competition, home, away, id });

  const maxGoals = 10;
  const pH = poissonPmfUpTo(base.lambdaHome, maxGoals);
  const pA = poissonPmfUpTo(base.lambdaAway, maxGoals);
  let pHomeWin = 0;
  let pDraw = 0;
  let pAwayWin = 0;
  for (let i = 0; i <= maxGoals; i += 1) {
    for (let j = 0; j <= maxGoals; j += 1) {
      const p = pH[i] * pA[j];
      if (i > j) pHomeWin += p;
      else if (i < j) pAwayWin += p;
      else pDraw += p;
    }
  }
  const s = pHomeWin + pDraw + pAwayWin;
  const raw1 = s > 0 ? pHomeWin / s : base.pHomeWin;
  const rawx = s > 0 ? pDraw / s : base.pDraw;
  const raw2 = s > 0 ? pAwayWin / s : base.pAwayWin;
  const px = clamp(rawx, 0.14, 0.34);
  const rem = clamp(1 - px, 1e-6, 0.999999);
  const sum12 = raw1 + raw2;
  const p1 = sum12 > 0 ? clamp((raw1 / sum12) * rem, 1e-6, 0.999999) : clamp(rem / 2, 1e-6, 0.999999);
  const p2 = sum12 > 0 ? clamp((raw2 / sum12) * rem, 1e-6, 0.999999) : clamp(rem / 2, 1e-6, 0.999999);
  return { lambdaHome: base.lambdaHome, lambdaAway: base.lambdaAway, pHomeWin: p1, pDraw: px, pAwayWin: p2 };
};

const computeSoccerCorrectScoreTopPrice = (args: {
  lambdaHome: number;
  lambdaAway: number;
  homeScore: number;
  awayScore: number;
  selection: string;
  overround: number;
}) => {
  const m = /^SCORE_(\d+)_(\d+)$/.exec(args.selection);
  if (!m) return null;
  const targetHome = Number(m[1]);
  const targetAway = Number(m[2]);
  if (!Number.isFinite(targetHome) || !Number.isFinite(targetAway)) return null;
  if (targetHome < args.homeScore || targetAway < args.awayScore) return null;

  const maxG = 6;
  const pH = poissonPmfUpTo(args.lambdaHome, maxG);
  const pA = poissonPmfUpTo(args.lambdaAway, maxG);
  const list: Array<{ home: number; away: number; p: number }> = [];
  for (let i = 0; i <= maxG; i += 1) {
    for (let j = 0; j <= maxG; j += 1) {
      const p = pH[i] * pA[j];
      list.push({ home: args.homeScore + i, away: args.awayScore + j, p });
    }
  }
  list.sort((a, b) => b.p - a.p);
  const top = list.slice(0, 9);
  const sumP = top.reduce((acc, x) => acc + x.p, 0);
  if (!(sumP > 0)) return null;
  const found = top.find((x) => x.home === targetHome && x.away === targetAway);
  if (!found || !(found.p > 0)) return null;

  const implied = clamp((found.p / sumP) * args.overround, 1e-9, 0.999999);
  return round2(clamp(1 / implied, 1.01, 100));
};

const findSoccerVirtualPricePrematch = (
  f: EventFixture,
  leg: { market: MarketKey; selection: SelectionKey; line?: number }
) => {
  const seedKey = `${f.home}:${f.away}:${f.competition}:${f.id}`;
  const model = computePrematchSoccerModel(seedKey);
  const overround = 1.07;

  const firstHalfProbs = () => {
    const maxGoals = 10;
    const lambdaHome = clamp(model.lambdaHome * 0.45, 0.05, 2.2);
    const lambdaAway = clamp(model.lambdaAway * 0.45, 0.05, 2.2);
    const pH = poissonPmfUpTo(lambdaHome, maxGoals);
    const pA = poissonPmfUpTo(lambdaAway, maxGoals);
    let pHomeWin = 0;
    let pDraw = 0;
    let pAwayWin = 0;
    for (let i = 0; i <= maxGoals; i += 1) {
      for (let j = 0; j <= maxGoals; j += 1) {
        const p = pH[i] * pA[j];
        if (i > j) pHomeWin += p;
        else if (i < j) pAwayWin += p;
        else pDraw += p;
      }
    }
    const s = pHomeWin + pDraw + pAwayWin;
    return {
      pHome: s > 0 ? pHomeWin / s : 1 / 3,
      pDraw: s > 0 ? pDraw / s : 1 / 3,
      pAway: s > 0 ? pAwayWin / s : 1 / 3
    };
  };

  if (leg.market === "FirstHalf1X2") {
    const p = firstHalfProbs();
    const wanted = leg.selection === "HOME" ? p.pHome : leg.selection === "DRAW" ? p.pDraw : leg.selection === "AWAY" ? p.pAway : null;
    if (wanted == null) return null;
    const implied = clamp(clamp(wanted, 1e-9, 0.999999) * overround, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line: undefined as number | undefined };
  }

  if (leg.market === "FirstHalfDoubleChance") {
    const p = firstHalfProbs();
    const val =
      leg.selection === "HOME_DRAW"
        ? p.pHome + p.pDraw
        : leg.selection === "HOME_AWAY"
          ? p.pHome + p.pAway
          : leg.selection === "DRAW_AWAY"
            ? p.pDraw + p.pAway
            : null;
    if (val == null) return null;
    const implied = clamp(clamp(val, 1e-9, 0.999999) * overround, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line: undefined as number | undefined };
  }

  if (leg.market === "FirstHalfDrawNoBet") {
    const p = firstHalfProbs();
    const denom = p.pHome + p.pAway;
    const val = denom > 0 ? (leg.selection === "HOME" ? p.pHome / denom : leg.selection === "AWAY" ? p.pAway / denom : null) : null;
    if (val == null) return null;
    const implied = clamp(clamp(val, 1e-9, 0.999999) * 1.06, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line: undefined as number | undefined };
  }

  if (leg.market === "FirstHalfOverUnder") {
    const line = typeof leg.line === "number" ? leg.line : null;
    if (line == null) return null;
    if (leg.selection !== "OVER" && leg.selection !== "UNDER") return null;
    const total = clamp((model.lambdaHome + model.lambdaAway) * 0.45, 0.1, 4.2);
    const under = poissonCdf(total, Math.floor(line));
    const over = 1 - under;
    const p = leg.selection === "OVER" ? over : under;
    const implied = clamp(clamp(p, 1e-9, 0.999999) * 1.06, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line };
  }

  if (leg.market === "DoubleChance") {
    const p1 = model.pHomeWin;
    const px = model.pDraw;
    const p2 = model.pAwayWin;
    const p =
      leg.selection === "HOME_DRAW"
        ? p1 + px
        : leg.selection === "HOME_AWAY"
          ? p1 + p2
          : leg.selection === "DRAW_AWAY"
            ? px + p2
            : null;
    if (p == null) return null;
    const implied = clamp(p * overround, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line: undefined as number | undefined };
  }

  if (leg.market === "DrawNoBet") {
    const p1 = model.pHomeWin;
    const p2 = model.pAwayWin;
    const denom = p1 + p2;
    const p = denom > 0 ? (leg.selection === "HOME" ? p1 / denom : leg.selection === "AWAY" ? p2 / denom : null) : null;
    if (p == null) return null;
    const implied = clamp(p * 1.06, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line: undefined as number | undefined };
  }

  if (leg.market === "BTTS") {
    const pYes = 1 - (Math.exp(-model.lambdaHome) + Math.exp(-model.lambdaAway) - Math.exp(-(model.lambdaHome + model.lambdaAway)));
    const p = leg.selection === "YES" ? pYes : leg.selection === "NO" ? 1 - pYes : null;
    if (p == null) return null;
    const implied = clamp(clamp(p, 1e-9, 0.999999) * 1.06, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line: undefined as number | undefined };
  }

  if (leg.market === "Handicap") {
    const line = typeof leg.line === "number" ? leg.line : null;
    if (line == null) return null;
    if (leg.selection !== "HOME" && leg.selection !== "AWAY") return null;

    const maxGoals = 10;
    const pH = poissonPmfUpTo(model.lambdaHome, maxGoals);
    const pA = poissonPmfUpTo(model.lambdaAway, maxGoals);
    let pHomeCover = 0;
    let pAwayCover = 0;
    let pPush = 0;
    for (let i = 0; i <= maxGoals; i += 1) {
      for (let j = 0; j <= maxGoals; j += 1) {
        const p = pH[i] * pA[j];
        const v = i + line - j;
        if (v > 0) pHomeCover += p;
        else if (v < 0) pAwayCover += p;
        else pPush += p;
      }
    }
    const denom = clamp(1 - pPush, 1e-9, 1);
    const p = leg.selection === "HOME" ? pHomeCover / denom : pAwayCover / denom;
    const implied = clamp(clamp(p, 1e-9, 0.999999) * 1.06, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line };
  }

  if (leg.market === "OverUnder") {
    const line = typeof leg.line === "number" ? leg.line : null;
    if (line == null) return null;
    const under = poissonCdf(model.lambdaHome + model.lambdaAway, Math.floor(line));
    const over = 1 - under;
    const p = leg.selection === "OVER" ? over : leg.selection === "UNDER" ? under : null;
    if (p == null) return null;
    const implied = clamp(clamp(p, 1e-9, 0.999999) * 1.06, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line };
  }

  if (leg.market === "CorrectScore") {
    const price = computeSoccerCorrectScoreTopPrice({
      lambdaHome: model.lambdaHome,
      lambdaAway: model.lambdaAway,
      homeScore: 0,
      awayScore: 0,
      selection: String(leg.selection),
      overround: 1.08
    });
    return price == null ? null : { price, line: undefined as number | undefined };
  }

  return null;
};

      const z = (-line - marginMean) / marginSd;
      const pHome = clamp(1 - normalCdf(z), 0.02, 0.98);
      const pAway = clamp(1 - pHome, 0.02, 0.98);
      const [oH, oA] = probsToOdds([pHome, pAway], 1.06);
      return { price: leg.selection === "HOME" ? oH : oA, line };
    }

    if (f.sport === "Hóquei") {
      const lambda = clamp(5.6 + (r(1) - 0.5) * 2.0, 3.2, 7.8);
      const split = clamp(0.52 + (r(2) - 0.5) * 0.18, 0.35, 0.7);
      const lambdaHome = lambda * split;
      const lambdaAway = lambda * (1 - split);
      const maxG = 10;
      const pH = poissonPmfUpTo(lambdaHome, maxG);
      const pA = poissonPmfUpTo(lambdaAway, maxG);
      let pHomeCover = 0;
      let pAwayCover = 0;
      let pPush = 0;
      for (let i = 0; i <= maxG; i += 1) {
        for (let j = 0; j <= maxG; j += 1) {
          const p = pH[i] * pA[j];
          const v = i + line - j;
          if (v > 0) pHomeCover += p;
          else if (v < 0) pAwayCover += p;
          else pPush += p;
        }
      }
      const denom = clamp(1 - pPush, 1e-9, 1);
      const p = leg.selection === "HOME" ? pHomeCover / denom : pAwayCover / denom;
      const implied = clamp(clamp(p, 1e-9, 0.999999) * 1.06, 1e-9, 0.999999);
      return { price: round2(clamp(1 / implied, 1.01, 100)), line };
    }

    if (f.sport === "Ténis") {
      const diffMean = clamp((r(9) - 0.5) * 5.0, -6, 6);
      const diffSd = clamp(3.2 + r(10) * 1.2, 2.6, 4.8);
      const z = (-line - diffMean) / diffSd;
      const pHome = clamp(1 - normalCdf(z), 0.02, 0.98);
      const pAway = clamp(1 - pHome, 0.02, 0.98);
      const [oH, oA] = probsToOdds([pHome, pAway], 1.06);
      return { price: leg.selection === "HOME" ? oH : oA, line };
    }

    return null;
  }

  if (leg.market === "FirstHalfOverUnder") {

    if (line == null) return null;
    if (leg.selection !== "OVER" && leg.selection !== "UNDER") return null;
    if (f.sport !== "Basquetebol") return null;

    const z = (line - mean1h) / sd1h;

  }

  }

  }

    const under = poissonCdf(lambdaP, Math.floor(line));
    const over = 1 - under;
    const [oOver, oUnder] = probsToOdds([clamp(over, 0.02, 0.98), clamp(under, 0.02, 0.98)], 1.06);
    return { price: leg.selection === "OVER" ? oOver : oUnder, line };
  }

  if (leg.market === "FirstPeriod1X2") {

    const pH = poissonPmfUpTo(lambdaHome, maxG);
    const pA = poissonPmfUpTo(lambdaAway, maxG);
    let pHomeWin = 0;
    let pDraw = 0;
    let pAwayWin = 0;
    for (let i = 0; i <= maxG; i += 1) {
      for (let j = 0; j <= maxG; j += 1) {

        if (i > j) pHomeWin += p;
        else if (i < j) pAwayWin += p;
        else pDraw += p;
      }
    }
    const s = pHomeWin + pDraw + pAwayWin;
    const p =
      s > 0
        ? leg.selection === "HOME"
          ? pHomeWin / s
          : leg.selection === "DRAW"
            ? pDraw / s
            : pAwayWin / s
        : null;
    if (p == null) return null;
    const implied = clamp(clamp(p, 1e-9, 0.999999) * 1.07, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line: undefined as number | undefined };

  if (leg.market === "Set1Winner") {

    return { price: leg.selection === "YES" ? oYes : oNo, line: undefined as number | undefined };
  }

  return null;
};

const hashString = (s: string) => {
  let h = 2166136261;
  for (let i = 0; i < s.length; i += 1) {
    h ^= s.charCodeAt(i);
    h = Math.imul(h, 16777619);
  }
  return h >>> 0;
};

const seeded = (seed: string) => {
  const h = hashString(seed);
  return (n: number) => {
    const x = Math.sin((h + n) * 9999) * 10000;
    return x - Math.floor(x);
  };
};

const getObj = (value: unknown): Record<string, unknown> | null => {
  if (!value || typeof value !== "object") return null;
  return value as Record<string, unknown>;
};

const getPath = (obj: unknown, path: string[]) => {
  let cur: unknown = obj;
  for (const k of path) {
    const o = getObj(cur);
    if (!o) return undefined;
    cur = o[k];
  }
  return cur;
};

const firstString = (obj: unknown, paths: string[][]) => {
  for (const p of paths) {
    const v = getPath(obj, p);
    if (typeof v === "string" && v.trim()) return v.trim();
    if (typeof v === "number" && Number.isFinite(v)) return String(v);
  }
  return "";
};

const toIso = (value: unknown) => {
  if (typeof value === "string" && value.trim()) {
    const d = new Date(value);
    if (!Number.isNaN(d.getTime())) return d.toISOString();
  }
  if (typeof value === "number" && Number.isFinite(value)) {
    const ms = value < 2_000_000_000 ? value * 1000 : value;
    const d = new Date(ms);
    if (!Number.isNaN(d.getTime())) return d.toISOString();
  }
  return "";
};

const extractArray = (json: unknown): unknown[] => {
  if (Array.isArray(json)) return json;
  const root = getObj(json);
  if (!root) return [];
  const candidates = ["data", "matches", "games", "events", "results", "livescores", "response"];
  for (const k of candidates) {
    const v = root[k];
    if (Array.isArray(v)) return v;
    const o = getObj(v);
    if (o && Array.isArray(o.data)) return o.data;
  }
  for (const v of Object.values(root)) {
    if (Array.isArray(v)) return v;
  }
  return [];
};

const parseDdMmYyyy = (dateStr: string) => {
  const m = /^(\d{2})\.(\d{2})\.(\d{4})$/.exec(dateStr.trim());
  if (!m) return null;
  const d = Number(m[1]);
  const mo = Number(m[2]);
  const y = Number(m[3]);
  if (!Number.isFinite(d) || !Number.isFinite(mo) || !Number.isFinite(y)) return null;
  return { d, mo, y };
};

const parseTimeHm = (timeStr: string) => {
  const m = /^(\d{1,2}):(\d{2})$/.exec(timeStr.trim());
  if (!m) return null;
  const h = Number(m[1]);
  const mi = Number(m[2]);
  if (!Number.isFinite(h) || !Number.isFinite(mi)) return null;
  return { h, mi };
};

const matchDateTimeToIso = (dateStr: string, timeStr: string) => {
  const d = parseDdMmYyyy(dateStr);
  const t = parseTimeHm(timeStr);
  if (!d || !t) return "";
  const iso = new Date(d.y, d.mo - 1, d.d, t.h, t.mi, 0).toISOString();
  return iso;
};

const canon = (s: string) =>
  s
    .normalize("NFD")
    .replace(/[\u0300-\u036f]/g, "")
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, " ")
    .trim()
    .replace(/\s+/g, " ");

const isYouthOrWomen = (text: string) => {
  const c = canon(text);
  if (!c) return false;
  if (/(^| )u ?(17|18|19|20)( |$)/.test(c)) return true;
  if (c.includes("women") || c.includes("womens") || c.includes("femin") || c.includes("femen")) return true;
  return false;
};

const isBlockedFixture = (competition: string, home: string, away: string) => {
  if (isYouthOrWomen(competition)) return true;
  if (isYouthOrWomen(home) || isYouthOrWomen(away)) return true;
  const h = home.trim().toLowerCase();
  const a = away.trim().toLowerCase();
  if (h.endsWith(" w") || a.endsWith(" w")) return true;
  if (h.includes("(w)") || a.includes("(w)")) return true;
  return false;
};

    .trim();
};

        const bySel = new Map(existing.odds.map((x) => [x.selection, x]));
        for (const x of merged) bySel.set(x.selection, x);
        existing.odds = Array.from(bySel.values());
      }
    };

    for (const om of oddsMarkets) {
      const marketObj = getObj(om);
      if (!marketObj) continue;
      const name = String(marketObj.name ?? marketObj.value ?? "").trim();
      const mk = marketFromName(name);
      if (!mk) continue;

      const booksNode = getPath(marketObj, ["bookmaker"]);
      const books = Array.isArray(booksNode) ? booksNode : booksNode ? [booksNode] : [];
      if (books.length === 0) continue;
      const book = pickBookmaker(books, mk.kind);
      if (!book) continue;

      const oddsArr = Array.isArray((book as Record<string, unknown>).odd) ? ((book as Record<string, unknown>).odd as unknown[]) : [];
      const totalArr = Array.isArray((book as Record<string, unknown>).total) ? ((book as Record<string, unknown>).total as unknown[]) : [];
      const handicapArr = Array.isArray((book as Record<string, unknown>).handicap) ? ((book as Record<string, unknown>).handicap as unknown[]) : [];

      if (mk.kind === "total" && totalArr.length) {
        for (const total of totalArr.slice(0, 28)) {
          const to0 = getObj(total);
          if (!to0) continue;
          const line = numberFromText(String(to0.name ?? ""));
          if (line == null) continue;
          if (!allowLine(mk.key, line)) continue;
          const oArr = Array.isArray((to0 as Record<string, unknown>).odd) ? ((to0 as Record<string, unknown>).odd as unknown[]) : [];
          const odds: Odds[] = [];
          for (const o of oArr) {
            const oo = getObj(o);
            if (!oo) continue;
            const sel = selectionFromOddName(String((oo as Record<string, unknown>).name ?? ""));
            const priceRaw = normalizeOdd((oo as Record<string, unknown>).value);
            if (!sel || priceRaw == null || !Number.isFinite(priceRaw)) continue;
            odds.push({ selection: sel, price: round2(priceRaw) });
          }
          addMarket(mk.key, odds, line);
        }
        continue;
      }

      if (mk.kind === "handicap" && handicapArr.length) {
        for (const h of handicapArr.slice(0, 28)) {
          const ho = getObj(h);
          if (!ho) continue;
          const lineRaw = signedNumberFromText(String(ho.name ?? "")) ?? numberFromText(String(ho.name ?? ""));
          const line = lineRaw == null ? null : Math.abs(lineRaw);
          if (line == null) continue;
          const oArr = Array.isArray((ho as Record<string, unknown>).odd) ? ((ho as Record<string, unknown>).odd as unknown[]) : [];
          const odds: Odds[] = [];
          for (const o of oArr) {
            const oo = getObj(o);
            if (!oo) continue;
            const sel = selectionFromOddName(String((oo as Record<string, unknown>).name ?? ""));
            const priceRaw = normalizeOdd((oo as Record<string, unknown>).value);
            if (!sel || priceRaw == null || !Number.isFinite(priceRaw)) continue;
            odds.push({ selection: sel, price: round2(priceRaw) });
          }
          addMarket(mk.key, odds, line);
        }
        continue;
      }

      if (oddsArr.length) {
        const odds: Odds[] = [];
        for (const o of oddsArr) {
          const oo = getObj(o);

          odds.push({ selection: sel, price: round2(priceRaw) });
        }
        addMarket(mk.key, odds);
      }
    }

    const markets = Array.from(marketMap.values()).filter((m) => (m.odds ?? []).length > 0);
    if (markets.length === 0) continue;

    const oneXTwo = markets.find((x) => x.key === "1X2") ?? null;
    const builtFallback = () => buildSoccerPrematchMarkets({ competition, home, away, id });
    const oddsOf = (sel: SelectionKey) => oneXTwo?.odds.find((o) => o.selection === sel)?.price ?? null;
    const sanitizeOneXTwo = () => {
      if (!oneXTwo) return;
      const h = oddsOf("HOME");
      const x = oddsOf("DRAW");
      const a = oddsOf("AWAY");
      const hasAll = h != null && x != null && a != null && Number.isFinite(h) && Number.isFinite(x) && Number.isFinite(a);
      const sum =
        hasAll && h && x && a
          ? (h > 0 ? 1 / h : 0) + (x > 0 ? 1 / x : 0) + (a > 0 ? 1 / a : 0)
          : 0;
      const looksBroken = !hasAll || sum < 0.9 || sum > 1.25 || (x != null && x > 15) || (h != null && h > 25) || (a != null && a > 25);
      if (!looksBroken) return;
      const built = builtFallback();
      oneXTwo.odds = built.oneXTwo;
    };

    if (!oneXTwo) {
      const built = builtFallback();
      markets.unshift({ key: "1X2", odds: built.oneXTwo });
    } else {
      sanitizeOneXTwo();
    }

    out.push({ id: `Futebol:${id}`, sport: "Futebol", competition, startsAt, home, away, markets });
  }

  return out;
};

const toFixture = (sport: EventFixture["sport"], raw: unknown): EventFixture | null => {
  const id =
    firstString(raw, [
      ["id"],
      ["match_id"],
      ["game_id"],
      ["fixture_id"],
      ["event_id"]
    ]) || nanoid();

  const home =
    firstString(raw, [
      ["home", "name"],
      ["home_team", "name"],
      ["homeTeam", "name"],
      ["localteam", "name"],
      ["teams", "home", "name"],
      ["team_home", "name"],
      ["home_name"],
      ["home"]
    ]) || "Home";

  const away =
    firstString(raw, [
      ["away", "name"],
      ["away_team", "name"],
      ["awayTeam", "name"],
      ["visitorteam", "name"],
      ["teams", "away", "name"],
      ["team_away", "name"],
      ["away_name"],
      ["away"]
    ]) || "Away";

  const competition =
    firstString(raw, [
      ["league", "name"],
      ["league_name"],
      ["competition", "name"],
      ["tournament", "name"],
      ["tournament_name"],
      ["league"],
      ["competition"]
    ]) || "Live";

  const startsAt =
    toIso(
      getPath(raw, ["startsAt"]) ??
        getPath(raw, ["start_time"]) ??
        getPath(raw, ["startTime"]) ??
        getPath(raw, ["starting_at"]) ??
        getPath(raw, ["time", "starting_at", "date_time"]) ??
        getPath(raw, ["date_time"]) ??
        getPath(raw, ["timestamp"]) ??
        getPath(raw, ["start_timestamp"])
    ) || nowPlusMinutes(10);

  if (isBlockedFixture(competition, home, away)) return null;

  const markets: EventFixture["markets"] = [];
  if (sport === "Futebol") {
    const built = buildSoccerPrematchMarkets({ competition, home, away, id });
    markets.push({
      key: "1X2",
      odds: built.oneXTwo
    });
    markets.push({
      key: "OverUnder",
      line: built.overUnder25.line,
      odds: built.overUnder25.odds
    });
  } else {
    const r = seeded(`${sport}:${competition}:${id}`);
    const pHome = clamp(0.52 + (r(1) - 0.5) * 0.18, 0.08, 0.92);
    const pAway = clamp(1 - pHome, 0.08, 0.92);
    const [home, away] = probsToOdds([pHome, pAway], 1.05);
    markets.push({
      key: "1X2",
      odds: [
        { selection: "HOME", price: home },
        { selection: "AWAY", price: away }
      ] as Odds[]
    });
  }

  return {
    id: `${sport}:${id}`,
    sport,
    competition,
    startsAt,
    home,
    away,
    markets
  };
};

    const types = Array.isArray(typeArr) ? typeArr : [];
    const marketMap = new Map<string, { key: MarketKey; line?: number; odds: Odds[] }>();

    const addMarket = (key: MarketKey, odds: Odds[], line?: number) => {
      const uniq = new Map<SelectionKey, Odds>();
      for (const it of odds) uniq.set(it.selection, it);
      const merged = Array.from(uniq.values()).filter((o) => Number.isFinite(o.price) && o.price > 1.0 && o.price < 200);
      if (merged.length === 0) return;
      const mapKey = `${key}:${line ?? ""}`;
      const existing = marketMap.get(mapKey);
      if (!existing) marketMap.set(mapKey, { key, line, odds: merged });
      else {
        const bySel = new Map(existing.odds.map((x) => [x.selection, x]));
        for (const x of merged) bySel.set(x.selection, x);
        existing.odds = Array.from(bySel.values());
      }
    };

    const pickBookmaker = (bookmakers: unknown[], want: "odd" | "total" | "handicap") => {
      const score = (b: unknown) => {
        const bo = getObj(b);
        if (!bo) return -1;
        const oddArr = Array.isArray((bo as Record<string, unknown>).odd) ? ((bo as Record<string, unknown>).odd as unknown[]) : [];
        const totalArr = Array.isArray((bo as Record<string, unknown>).total) ? ((bo as Record<string, unknown>).total as unknown[]) : [];
        const handicapArr = Array.isArray((bo as Record<string, unknown>).handicap) ? ((bo as Record<string, unknown>).handicap as unknown[]) : [];
        if (want === "total") return totalArr.length * 10 + oddArr.length;
        if (want === "handicap") return handicapArr.length * 10 + oddArr.length;
        return oddArr.length;
      };
      const sorted = [...bookmakers].sort((a, b) => score(b) - score(a));
      return getObj(sorted[0]);
    };

    for (const t of types) {
      const to = getObj(t);
      if (!to) continue;
      const typeValue = String(to.value ?? to.name ?? to.type ?? to.id ?? "").trim();
      const key = marketFromTypeValue(typeValue);
      if (!key) continue;

      const booksNode = getPath(to, ["bookmaker"]);
      const books = Array.isArray(booksNode) ? booksNode : booksNode ? [booksNode] : [];
      if (books.length === 0) continue;

      const want: "odd" | "total" | "handicap" =
        key === "OverUnder" || key === "FirstHalfOverUnder" || key === "FirstQuarterOverUnder" || key === "TeamTotalHome" || key === "TeamTotalAway"
          ? "total"
          : key === "Handicap"
            ? "handicap"
            : "odd";
      const book = pickBookmaker(books, want);
      if (!book) continue;

      const oddsArr = Array.isArray((book as Record<string, unknown>).odd) ? ((book as Record<string, unknown>).odd as unknown[]) : [];
      const totalArr = Array.isArray((book as Record<string, unknown>).total) ? ((book as Record<string, unknown>).total as unknown[]) : [];
      const handicapArr = Array.isArray((book as Record<string, unknown>).handicap) ? ((book as Record<string, unknown>).handicap as unknown[]) : [];

      const isTotalLike =
        key === "OverUnder" ||
        key === "FirstHalfOverUnder" ||
        key === "FirstQuarterOverUnder" ||
        key === "TeamTotalHome" ||
        key === "TeamTotalAway";

      if (isTotalLike && totalArr.length) {
        for (const total of totalArr.slice(0, 28)) {
          const to0 = getObj(total);
          if (!to0) continue;
          const line = numberFromText(String(to0.name ?? ""));
          if (line == null) continue;
          const oArr = Array.isArray((to0 as Record<string, unknown>).odd) ? ((to0 as Record<string, unknown>).odd as unknown[]) : [];
          const odds: Odds[] = [];
          for (const o of oArr) {
            const oo = getObj(o);
            if (!oo) continue;
            const sel = selectionFromOddName(String((oo as Record<string, unknown>).name ?? ""));
            const priceRaw = normalizeOdd((oo as Record<string, unknown>).value);
            if (!sel || priceRaw == null || !Number.isFinite(priceRaw)) continue;
            odds.push({ selection: sel, price: round2(priceRaw) });
          }
          addMarket(key, odds, line);
        }
        continue;
      }

      if (key === "Handicap" && handicapArr.length) {
        for (const h of handicapArr.slice(0, 28)) {
          const ho = getObj(h);
          if (!ho) continue;
          const line = signedNumberFromText(String(ho.name ?? "")) ?? numberFromText(String(ho.name ?? ""));
          if (line == null) continue;
          const oArr = Array.isArray((ho as Record<string, unknown>).odd) ? ((ho as Record<string, unknown>).odd as unknown[]) : [];
          const odds: Odds[] = [];
          for (const o of oArr) {
            const oo = getObj(o);
            if (!oo) continue;
            const sel = selectionFromOddName(String((oo as Record<string, unknown>).name ?? ""));
            const priceRaw = normalizeOdd((oo as Record<string, unknown>).value);
            if (!sel || priceRaw == null || !Number.isFinite(priceRaw)) continue;
            odds.push({ selection: sel, price: round2(priceRaw) });
          }
          addMarket(key, odds, line);
        }
        continue;
      }

      if (oddsArr.length) {
        const odds: Odds[] = [];
        for (const o of oddsArr) {
          const oo = getObj(o);
          if (!oo) continue;
          const sel = selectionFromOddName(String((oo as Record<string, unknown>).name ?? ""));
          const priceRaw = normalizeOdd((oo as Record<string, unknown>).value);
          if (!sel || priceRaw == null || !Number.isFinite(priceRaw)) continue;
          odds.push({ selection: sel, price: round2(priceRaw) });
        }
        addMarket(key, odds);
      }
    }

    const markets = Array.from(marketMap.values()).filter((m) => (m.odds ?? []).length > 0);
    if (markets.length === 0) continue;
    out.push({ id: `Basquetebol:${id}`, sport: "Basquetebol", competition, startsAt, home, away, markets });
  }

  return out;
};

const tennisCompetitionFromTournamentName = (name: string) => {
  const c = canon(name);
  if (c.includes("australian open") || c.includes("roland garros") || c.includes("wimbledon") || c.includes("us open")) return "Grand Slams";
  if (c.includes("wta")) return "WTA Tour";
  if (c.includes("atp")) return "ATP Tour";
  return "ATP Tour";
};

const numberFromText = (text: string) => {
  const cleaned = String(text).replace(/,/g, ".");
  const m = /(-?\d+(?:\.\d+)?)/.exec(cleaned);
  if (!m) return null;
  const n = Number(m[1]);
  return Number.isFinite(n) ? n : null;
};

const signedNumberFromText = (text: string) => {
  const cleaned = String(text).replace(/,/g, ".");
  const m = /([+-]\s*\d+(?:\.\d+)?)/.exec(cleaned);
  if (!m) return null;
  const n = Number(m[1].replace(/\s+/g, ""));
  return Number.isFinite(n) ? n : null;
};

const tennisMarketFromTypeName = (typeNameRaw: string): { key: MarketKey; line?: number } | null => {
  const n = canon(typeNameRaw);
  if (!n) return null;
  if (n.includes("half") || n.includes("goal") || n.includes("corner") || n.includes("card")) return null;
  if (n.includes("correct") && n.includes("score")) return { key: "CorrectScore" };
  if (n.includes("set betting")) return { key: "CorrectScore" };
  if ((n.includes("win") && n.includes("set")) || n.includes("to win a set") || n.includes("wins a set") || n.includes("at least one set") || n.includes("win 1 set"))
    return { key: "WinAtLeastOneSet" };
  if ((n.includes("1st set") || n.includes("set 1")) && (n.includes("over") || n.includes("under") || n.includes("total"))) {
    const line = numberFromText(typeNameRaw);
    return line != null ? { key: "Set1GamesOverUnder", line } : { key: "Set1GamesOverUnder" };
  }
  if ((n.includes("2nd set") || n.includes("set 2")) && (n.includes("over") || n.includes("under") || n.includes("total"))) {
    const line = numberFromText(typeNameRaw);
    return line != null ? { key: "Set2GamesOverUnder", line } : { key: "Set2GamesOverUnder" };
  }
  if ((n.includes("3rd set") || n.includes("set 3")) && (n.includes("over") || n.includes("under") || n.includes("total"))) {
    const line = numberFromText(typeNameRaw);
    return line != null ? { key: "Set3GamesOverUnder", line } : { key: "Set3GamesOverUnder" };
  }
  if (n.includes("1st set") || n.includes("set 1")) return { key: "Set1Winner" };
  if (n.includes("2nd set") || n.includes("set 2")) return { key: "Set2Winner" };
  if (n.includes("3rd set") || n.includes("set 3")) return { key: "Set3Winner" };
  if (n.includes("aces") && (n.includes("over") || n.includes("under") || n.includes("total"))) {
    const line = numberFromText(typeNameRaw);
    return line != null ? { key: "AcesOverUnder", line } : { key: "AcesOverUnder" };
  }
  if (n.includes("total sets") && (n.includes("over") || n.includes("under") || n.includes("total"))) {
    const line = numberFromText(typeNameRaw);
    return line != null ? { key: "TotalSetsOverUnder", line } : { key: "TotalSetsOverUnder" };
  }
  if ((n.includes("tie") && n.includes("break")) || n.includes("tiebreak")) return { key: "TieBreak" };
  if (n.includes("handicap")) {
    const line = signedNumberFromText(typeNameRaw) ?? numberFromText(typeNameRaw);
    return line != null ? { key: "Handicap", line } : { key: "Handicap" };
  }
  if ((n.includes("games handicap") || n.includes("handicap games")) && !n.includes("set")) {
    const line = signedNumberFromText(typeNameRaw) ?? numberFromText(typeNameRaw);
    return line != null ? { key: "Handicap", line } : { key: "Handicap" };
  }
  if (n.includes("games") && (n.includes("over") || n.includes("under") || n.includes("total"))) {
    const line = numberFromText(typeNameRaw);
    return line != null ? { key: "OverUnder", line } : { key: "OverUnder" };
  }
  if (n.includes("match") && (n.includes("winner") || n.includes("win"))) return { key: "1X2" };
  if (n.includes("moneyline") || (n.includes("winner") && !n.includes("set"))) return { key: "1X2" };
  if (n.includes("home") && n.includes("away")) return { key: "1X2" };
  return null;
};

const tennisSelectionFromOddName = (oddNameRaw: string, home: string, away: string): "HOME" | "AWAY" | null => {
  const n = canon(oddNameRaw);
  if (!n) return null;
  const h = canon(home);
  const a = canon(away);
  if (h && n.includes(h)) return "HOME";
  if (a && n.includes(a)) return "AWAY";
  if (n.includes("home") || n.includes("player 1") || n.includes("player1") || n.includes("p1") || n === "1") return "HOME";
  if (n.includes("away") || n.includes("player 2") || n.includes("player2") || n.includes("p2") || n === "2") return "AWAY";
  return null;
};

const tennisCorrectScoreFromOddName = (oddNameRaw: string) => {
  const m = /(\d)\s*[-:]\s*(\d)/.exec(String(oddNameRaw));
  if (!m) return null;
  const h = Number(m[1]);
  const a = Number(m[2]);
  if (!Number.isFinite(h) || !Number.isFinite(a)) return null;
  if (h < 0 || a < 0 || h > 3 || a > 3) return null;
  return { h, a };
};

const isTennisSetCompleted = (s: { h: number; a: number }) => {
  const h = Math.max(0, Math.floor(s.h));
  const a = Math.max(0, Math.floor(s.a));
  const hi = Math.max(h, a);
  const lo = Math.min(h, a);
  if (hi < 6) return false;
  if (hi === 6) return hi - lo >= 2;
  if (hi === 7) return lo === 5 || lo === 6;
  return true;
};

const parseTennisSetPairsFromText = (text: string) => {
  const out: Array<{ h: number; a: number }> = [];
  const rx = /(\d{1,2})\s*[-:]\s*(\d{1,2})/g;
  let m: RegExpExecArray | null = null;
  while ((m = rx.exec(text))) {
    const h = Number(m[1]);
    const a = Number(m[2]);
    if (!Number.isFinite(h) || !Number.isFinite(a)) continue;
    if (h < 0 || a < 0) continue;
    if (h > 7 || a > 7) continue;
    out.push({ h, a });
    if (out.length >= 5) break;
  }
  return out;
};

const collectDeepStrings = (value: unknown, out: string[], depth: number) => {
  if (out.length >= 30) return;
  if (depth <= 0) return;
  if (typeof value === "string") {
    const s = value.trim();
    if (s && s.length <= 80 && /\d/.test(s) && /[-:]/.test(s)) out.push(s);
    return;
  }
  if (Array.isArray(value)) {
    for (const v of value) collectDeepStrings(v, out, depth - 1);
    return;
  }
  const o = getObj(value);
  if (!o) return;
  for (const v of Object.values(o)) collectDeepStrings(v, out, depth - 1);
};

const collectDeepTennisPairArrays = (value: unknown, out: Array<Array<{ h: number; a: number }>>, depth: number, keyHint: string) => {
  if (out.length >= 10) return;
  if (depth <= 0) return;
  if (Array.isArray(value)) {
    if (value.length >= 1 && value.length <= 5) {
      const pairs: Array<{ h: number; a: number }> = [];
      for (const el of value) {
        const o = getObj(el);
        if (!o) {
          pairs.length = 0;
          break;
        }
        const read = (k1: string, k2: string) => {
          const n1 = Number((o as Record<string, unknown>)[k1] ?? "");
          const n2 = Number((o as Record<string, unknown>)[k2] ?? "");
          return Number.isFinite(n1) && Number.isFinite(n2) ? { h: n1, a: n2 } : null;
        };
        const p =
          read("home", "away") ??
          read("h", "a") ??
          read("player1", "player2") ??
          read("p1", "p2") ??
          read("team1", "team2") ??
          read("1", "2");
        if (!p) {
          pairs.length = 0;
          break;
        }
        const h = Math.max(0, Math.floor(p.h));
        const a = Math.max(0, Math.floor(p.a));
        if (h > 7 || a > 7) {
          pairs.length = 0;
          break;
        }
        pairs.push({ h, a });
      }
      const k = keyHint.toLowerCase();
      if (pairs.length && (k.includes("set") || k.includes("score") || k.includes("period"))) out.push(pairs);
    }
    for (const v of value) collectDeepTennisPairArrays(v, out, depth - 1, keyHint);
    return;
  }
  const o = getObj(value);
  if (!o) return;
  for (const [k, v] of Object.entries(o)) collectDeepTennisPairArrays(v, out, depth - 1, k);
};

const tennisLiveFromMatch = (mo: Record<string, unknown>): TennisLiveState | null => {
  const statusRaw = String(mo.status ?? mo.status_desc ?? mo.statusText ?? mo.state ?? "").trim();
  const status = statusRaw.toLowerCase();
  const finishedByStatus =
    status.includes("finished") || status.includes("retired") || status.includes("walkover") || status === "ft" || status === "final";

  const candidates: string[] = [];
  collectDeepStrings(mo, candidates, 5);
  const primaryStrings: string[] = [String(mo.score ?? ""), String(mo.ss ?? ""), String(mo.result ?? ""), String(mo.final_score ?? ""), String(mo.set_score ?? "")]
    .map((s) => s.trim())
    .filter(Boolean);
  const allStrings = [...primaryStrings, ...candidates];

  let best: Array<{ h: number; a: number }> = [];
  let bestScore = -1;
  for (const s of allStrings) {
    const pairs = parseTennisSetPairsFromText(s);
    if (pairs.length === 0) continue;
    const hasSetLike = pairs.some((p) => Math.max(p.h, p.a) >= 4);
    const score = pairs.length * 10 + (hasSetLike ? 3 : 0);
    if (score > bestScore) {
      bestScore = score;
      best = pairs;
    }
  }

  if (best.length === 0) {
    const arrays: Array<Array<{ h: number; a: number }>> = [];
    collectDeepTennisPairArrays(mo, arrays, 6, "");
    arrays.sort((a, b) => b.length - a.length);
    best = arrays[0] ?? [];
  }

  if (best.length === 0) return null;

  const last = best[best.length - 1] ?? { h: 0, a: 0 };
  const completedSets = best.filter(isTennisSetCompleted).slice(0, 5);
  const lastCompleted = isTennisSetCompleted(last);
  const current = lastCompleted ? { h: 0, a: 0 } : last;

  let winsHome = 0;
  let winsAway = 0;
  for (const s of completedSets) {
    if (s.h > s.a) winsHome += 1;
    else if (s.a > s.h) winsAway += 1;
  }

  const finishedBySets = winsHome === 2 || winsAway === 2;
  const finished = finishedByStatus || finishedBySets;
  const bestLen = Math.max(0, Math.min(5, best.length));
  const currentSet = finished ? Math.max(1, completedSets.length) : Math.min(5, Math.max(completedSets.length + 1, bestLen || 1));
  const clock = finished ? "FINAL" : `S${currentSet}`;

  return {
    clock,
    currentSet,
    current,
    completedSets,
    finished,
    status: statusRaw || undefined,
    updatedAt: Date.now()
  };
};

const tennisSetCellToInt = (value: unknown) => {
  const s = String(value ?? "").trim();
  if (!s) return null;
  const n = Number(s.replace(/,/g, "."));
  if (!Number.isFinite(n)) return null;
  const intPart = Math.max(0, Math.floor(n));
  if (intPart > 7) return null;
  return intPart;
};

const tennisLiveFromLivescoreMatch = (mo: Record<string, unknown>): TennisLiveState | null => {
  const statusRaw = String(mo.status ?? "").trim();
  const status = statusRaw.toLowerCase();
  const finished = status.includes("finished") || status.includes("retired") || status.includes("walkover");
  const setFromStatus = (() => {
    const m = /set\s*(\d+)/i.exec(statusRaw);
    const n = m ? Number(m[1]) : NaN;
    return Number.isFinite(n) ? Math.max(1, Math.min(5, Math.floor(n))) : null;
  })();

  const playersNode = getPath(mo, ["player"]);
  const players = Array.isArray(playersNode) ? playersNode : playersNode ? [playersNode] : [];
  if (players.length < 2) return null;
  const p0 = getObj(players[0]) ?? {};
  const p1 = getObj(players[1]) ?? {};

  const sets: Array<{ h: number; a: number }> = [];
  for (let i = 1; i <= 5; i += 1) {
    const h = tennisSetCellToInt((p0 as Record<string, unknown>)[`s${i}`]);
    const a = tennisSetCellToInt((p1 as Record<string, unknown>)[`s${i}`]);
    if (h == null || a == null) continue;
    sets.push({ h, a });
  }

  const completedSets = sets.filter(isTennisSetCompleted).slice(0, 5);
  const currentSet = finished
    ? Math.max(1, completedSets.length)
    : setFromStatus != null
      ? setFromStatus
      : Math.min(5, Math.max(completedSets.length + 1, sets.length || 1));

  const currentPair = sets[currentSet - 1] ?? { h: 0, a: 0 };
  const current = finished || isTennisSetCompleted(currentPair) ? { h: 0, a: 0 } : currentPair;
  const clock = finished ? "FINAL" : `S${currentSet}`;

  return {
    clock,
    currentSet,
    current,
    completedSets,
    finished,
    status: statusRaw || undefined,
    updatedAt: Date.now()
  };
};

const parseTennisOddsFixtures = (json: unknown): EventFixture[] => {
  const tArr = getPath(json, ["odds", "tournament"]);
  const tournaments = Array.isArray(tArr) ? tArr : [];
  const out: EventFixture[] = [];

  for (const t of tournaments) {
    const to = getObj(t);
    if (!to) continue;
    const tournamentName = String(to.name ?? to.league ?? "Ténis").trim() || "Ténis";
    const competition = tennisCompetitionFromTournamentName(tournamentName);

    const matchNode = getPath(to, ["matches", "match"]);
    const matches = Array.isArray(matchNode) ? matchNode : matchNode ? [matchNode] : [];

    for (const m of matches) {
      const mo = getObj(m);
      if (!mo) continue;
      const statusRaw = String(mo.status ?? "").trim().toLowerCase();
      if (statusRaw.includes("finished") || statusRaw.includes("retired") || statusRaw.includes("walkover")) continue;
      if (statusRaw === "0") continue;

      const id = String(mo.id ?? nanoid()).trim() || nanoid();
      const dateStr = String(mo.date ?? "").trim();
      const timeStr = String(mo.time ?? "").trim();
      const startsAt = matchDateTimeToIso(dateStr, timeStr) || nowPlusMinutes(45);

      const playersNode = getPath(mo, ["player"]);
      const players = Array.isArray(playersNode) ? playersNode : playersNode ? [playersNode] : [];
      if (players.length < 2) continue;
      const p0 = getObj(players[0]) ?? {};
      const p1 = getObj(players[1]) ?? {};
      const home = String(p0.name ?? "Home").trim() || "Home";
      const away = String(p1.name ?? "Away").trim() || "Away";
      if (isBlockedFixture(competition, home, away)) continue;

      const typeArr = getPath(mo, ["odds", "type"]);
      const types = Array.isArray(typeArr) ? typeArr : typeArr ? [typeArr] : [];
      const marketMap = new Map<string, { key: MarketKey; line?: number; odds: Odds[] }>();

      for (const t of types) {
        const to = getObj(t);
        if (!to) continue;
        const typeName = String(to.name ?? to.type ?? to.value ?? to.id ?? "").trim();
        const mk = tennisMarketFromTypeName(typeName);
        if (!mk) continue;
        const booksNode = getPath(to, ["bookmaker"]);
        const books = Array.isArray(booksNode) ? booksNode : booksNode ? [booksNode] : [];

        const scoreBook = (b: unknown) => {
          const bo = getObj(b);
          if (!bo) return -1;
          const totalArr = Array.isArray((bo as Record<string, unknown>).total) ? ((bo as Record<string, unknown>).total as unknown[]) : [];
          const handicapArr = Array.isArray((bo as Record<string, unknown>).handicap) ? ((bo as Record<string, unknown>).handicap as unknown[]) : [];
          const oddArr = Array.isArray((bo as Record<string, unknown>).odd) ? ((bo as Record<string, unknown>).odd as unknown[]) : [];
          if (
            mk.key === "OverUnder" ||
            mk.key === "Set1GamesOverUnder" ||
            mk.key === "Set2GamesOverUnder" ||
            mk.key === "Set3GamesOverUnder" ||
            mk.key === "AcesOverUnder" ||
            mk.key === "TotalSetsOverUnder"
          )
            return totalArr.length * 10 + oddArr.length;
          if (mk.key === "Handicap") return handicapArr.length * 10 + oddArr.length;
          return oddArr.length;
        };

        books.sort((a, b) => scoreBook(b) - scoreBook(a));
        const bestBook = getObj(books[0]);
        if (!bestBook) continue;

        const oddArr = Array.isArray((bestBook as Record<string, unknown>).odd) ? ((bestBook as Record<string, unknown>).odd as unknown[]) : [];
        const totalArr = Array.isArray((bestBook as Record<string, unknown>).total) ? ((bestBook as Record<string, unknown>).total as unknown[]) : [];
        const handicapArr = Array.isArray((bestBook as Record<string, unknown>).handicap) ? ((bestBook as Record<string, unknown>).handicap as unknown[]) : [];

        const addMarket = (m: { key: MarketKey; line?: number; odds: Odds[] }) => {
          const key = `${m.key}:${m.line ?? ""}`;
          const existing = marketMap.get(key);
          if (!existing) {
            marketMap.set(key, m);
            return;
          }
          const bySel = new Map(existing.odds.map((x) => [x.selection, x]));
          for (const x of m.odds) bySel.set(x.selection, x);
          existing.odds = Array.from(bySel.values());
          if (existing.line == null && m.line != null) existing.line = m.line;
        };

        const parseOddItems = (items: unknown[], lineDefault?: number) => {
          let line: number | undefined = lineDefault;
          const oddsOut: Odds[] = [];
          for (const o of items) {
            const oo = getObj(o);
            if (!oo) continue;
            const nameRaw = String(oo.name ?? "").trim();
            const name = canon(nameRaw);
            const priceRaw = Number(oo.value ?? "");
            if (!Number.isFinite(priceRaw)) continue;
            const price = round2(priceRaw);
            if (!Number.isFinite(price) || price <= 1.0 || price >= 200) continue;

            if (mk.key === "CorrectScore") {
              const cs = tennisCorrectScoreFromOddName(nameRaw);
              if (!cs) continue;
              oddsOut.push({ selection: `SCORE_${cs.h}_${cs.a}` as SelectionKey, price });
              continue;
            }

            if (mk.key === "1X2" || mk.key === "Set1Winner" || mk.key === "Set2Winner" || mk.key === "Set3Winner" || mk.key === "WinAtLeastOneSet") {
              const sel = tennisSelectionFromOddName(nameRaw, home, away);
              if (!sel) continue;
              oddsOut.push({ selection: sel, price });
              continue;
            }

            if (mk.key === "Handicap") {
              const sel = tennisSelectionFromOddName(nameRaw, home, away);
              if (!sel) continue;
              if (line == null) {
                const ln = signedNumberFromText(nameRaw) ?? signedNumberFromText(typeName) ?? numberFromText(nameRaw) ?? numberFromText(typeName);
                if (ln != null) line = ln;
              }
              oddsOut.push({ selection: sel, price });
              continue;
            }

            if (
              mk.key === "OverUnder" ||
              mk.key === "Set1GamesOverUnder" ||
              mk.key === "Set2GamesOverUnder" ||
              mk.key === "Set3GamesOverUnder" ||
              mk.key === "AcesOverUnder" ||
              mk.key === "TotalSetsOverUnder"
            ) {
              const isOver = name.includes("over") || name.startsWith("o ");
              const isUnder = name.includes("under") || name.startsWith("u ");
              if (!isOver && !isUnder) continue;
              if (line == null) {
                const ln = numberFromText(nameRaw) ?? numberFromText(typeName);
                if (ln != null) line = ln;
              }
              oddsOut.push({ selection: isOver ? "OVER" : "UNDER", price });
              continue;
            }

            if (mk.key === "TieBreak") {
              const yes = name.includes("yes") || name.includes("sim") || name === "y";
              const no = name.includes("no") || name.includes("nao") || name === "n";
              if (!yes && !no) continue;
              oddsOut.push({ selection: yes ? "YES" : "NO", price });
              continue;
            }
          }

          const uniq = new Map<SelectionKey, Odds>();
          for (const it of oddsOut) uniq.set(it.selection, it);
          const merged = Array.from(uniq.values());
          return { line, odds: merged };
        };

        const isTotalLike =
          mk.key === "OverUnder" ||
          mk.key === "Set1GamesOverUnder" ||
          mk.key === "Set2GamesOverUnder" ||
          mk.key === "Set3GamesOverUnder" ||
          mk.key === "AcesOverUnder" ||
          mk.key === "TotalSetsOverUnder";

        if (isTotalLike && totalArr.length) {
          const maxLines = 28;
          for (const t0 of totalArr.slice(0, maxLines)) {
            const to0 = getObj(t0);
            if (!to0) continue;
            const line0 = numberFromText(String(to0.name ?? ""));
            if (line0 == null) continue;
            const oArr = Array.isArray((to0 as Record<string, unknown>).odd) ? ((to0 as Record<string, unknown>).odd as unknown[]) : [];
            const parsed = parseOddItems(oArr, line0);
            if (parsed.odds.length === 0) continue;
            addMarket({ key: mk.key, line: parsed.line, odds: parsed.odds });
          }
          continue;
        }

        if (mk.key === "Handicap" && handicapArr.length) {
          const maxLines = 28;
          for (const h0 of handicapArr.slice(0, maxLines)) {
            const ho = getObj(h0);
            if (!ho) continue;
            const line0 = signedNumberFromText(String(ho.name ?? "")) ?? numberFromText(String(ho.name ?? ""));
            if (line0 == null) continue;
            const oArr = Array.isArray((ho as Record<string, unknown>).odd) ? ((ho as Record<string, unknown>).odd as unknown[]) : [];
            const parsed = parseOddItems(oArr, line0);
            if (parsed.odds.length === 0) continue;
            addMarket({ key: mk.key, line: parsed.line, odds: parsed.odds });
          }
          continue;
        }

        if (oddArr.length) {
          const parsed = parseOddItems(oddArr, mk.line);
          if (parsed.odds.length === 0) continue;
          if (
            (mk.key === "OverUnder" ||
              mk.key === "Set1GamesOverUnder" ||
              mk.key === "Set2GamesOverUnder" ||
              mk.key === "Set3GamesOverUnder" ||
              mk.key === "AcesOverUnder" ||
              mk.key === "TotalSetsOverUnder" ||
              mk.key === "Handicap") &&
            parsed.line == null
          )
            continue;
          addMarket({ key: mk.key, line: parsed.line, odds: parsed.odds });
        }
      }

      const markets = Array.from(marketMap.values()).filter((m) => (m.odds ?? []).length > 0);
      if (markets.length === 0) continue;

      const live = tennisLiveFromMatch(mo);
      out.push({
        id: `Ténis:${id}`,
        sport: "Ténis",
        competition,
        startsAt,
        home,
        away,
        markets,
        live: live ? { tennis: live } : undefined
      });
    }
  }

  return out;
};

const applyTennisLivescoresToFixtures = (fixtures: EventFixture[], livescoresJson: unknown) => {
  const map = new Map<string, TennisLiveState>();
  const addMatchesFromContainer = (container: unknown) => {
    const co = getObj(container);
    if (!co) return;
    const matchNode = getPath(co, ["match"]) ?? getPath(co, ["matches", "match"]);
    const matches = Array.isArray(matchNode) ? matchNode : matchNode ? [matchNode] : [];
    for (const m of matches) {
      const mo = getObj(m);
      if (!mo) continue;
      const id = String(mo.id ?? "").trim();
      if (!id) continue;
      const live = tennisLiveFromLivescoreMatch(mo);
      if (!live) continue;
      map.set(id, live);
    }
  };

  const tourNode = getPath(livescoresJson, ["livescores", "tournament"]);
  const tours = Array.isArray(tourNode) ? tourNode : tourNode ? [tourNode] : [];
  for (const t of tours) addMatchesFromContainer(t);

  const catNode = getPath(livescoresJson, ["livescores", "category"]);
  const cats = Array.isArray(catNode) ? catNode : catNode ? [catNode] : [];
  for (const c of cats) addMatchesFromContainer(c);

  if (map.size === 0) return fixtures;
  return fixtures.map((f) => {
    if (f.sport !== "Ténis") return f;
    const rawId = String(f.id ?? "");
    const id = rawId.startsWith("Ténis:") ? rawId.slice("Ténis:".length) : rawId;
    const live = map.get(id);
    if (!live) return f;
    return { ...f, live: { ...(f.live ?? {}), tennis: live } };
  });
};

type StatPalCached = { ok: boolean; status: number; json: unknown; fetchedAt: number; nextAttemptAt: number; rateLimited: boolean };
const statpalCache = new Map<string, StatPalCached>();
type StatPalBinaryCached = { ok: boolean; status: number; body: Buffer; contentType: string; fetchedAt: number; nextAttemptAt: number; rateLimited: boolean };
const statpalBinaryCache = new Map<string, StatPalBinaryCached>();

const queryValueToString = (value: unknown) => {
  if (typeof value === "string") return value;
  if (Array.isArray(value) && typeof value[0] === "string") return value[0];
  if (typeof value === "number" && Number.isFinite(value)) return String(value);
  return "";
};

const buildStatPalUrlWithKey = (pathname: string, query: Record<string, unknown>, accessKey: string) => {
  const u = new URL(`${STATPAL_BASE_URL}${pathname}`);
  for (const [k, v] of Object.entries(query)) {
    if (!k || k === "access_key") continue;
    const s = queryValueToString(v);
    if (!s) continue;
    u.searchParams.set(k, s);
  }
  u.searchParams.set("access_key", accessKey);
  return u.toString();
};

const buildStatPalUrl = (pathname: string, query: Record<string, unknown>) => {
  return buildStatPalUrlWithKey(pathname, query, STATPAL_ACCESS_KEY);
};

const statpalFetchJson = async (url: string, controller: AbortController) => {
  const res = await fetch(url, { signal: controller.signal });
  const text = await res.text();
  let json: unknown = null;
  try {
    json = JSON.parse(text) as unknown;
  } catch {
    json = null;
  }
  const root = getObj(json);
  const rateLimited = Boolean(
    res.status === 429 ||
      (root != null && typeof root.error === "string" && root.error.toLowerCase().includes("rate limit")) ||
      (root != null && typeof root.message === "string" && root.message.toLowerCase().includes("rate limit"))
  );
  return { ok: res.ok, status: res.status, json, rateLimited };
};

const statpalGetCachedTryKeys = async (
  pathname: string,
  query: Record<string, unknown>,
  ttlMs: number,
  backoffOnRateLimitMs: number,
  preferKey?: string
) => {
  const keysBase = STATPAL_ACCESS_KEYS.length ? STATPAL_ACCESS_KEYS : STATPAL_ACCESS_KEY ? [STATPAL_ACCESS_KEY] : [];
  const keys = preferKey && keysBase.includes(preferKey) ? [preferKey, ...keysBase.filter((k) => k !== preferKey)] : keysBase;
  let last: StatPalCached = { ok: false, status: 0, json: null, fetchedAt: Date.now(), nextAttemptAt: 0, rateLimited: false };
  for (const k of keys) {
    const url = buildStatPalUrlWithKey(pathname, query, k);
    const fetched = await statpalGetCached(url, ttlMs, backoffOnRateLimitMs);
    last = fetched;
    if (fetched.ok) return fetched;
    if (!fetched.rateLimited) continue;
  }
  return last;
};

const statpalFetchBinary = async (url: string, controller: AbortController, accept: string) => {
  const res = await fetch(url, { signal: controller.signal, headers: { Accept: accept } });
  const ab = await res.arrayBuffer();
  const body = Buffer.from(ab);
  const contentType = String(res.headers.get("content-type") ?? "");
  let rateLimited = res.status === 429;
  if (!rateLimited && contentType.includes("application/json")) {
    try {
      const json = JSON.parse(body.toString("utf8")) as unknown;
      const root = getObj(json);
      rateLimited = Boolean(
        root != null && typeof root.error === "string" && root.error.toLowerCase().includes("rate limit")
      );
    } catch {
      rateLimited = false;
    }
  }
  return { ok: res.ok, status: res.status, body, contentType, rateLimited };
};

const statpalGetCached = async (url: string, ttlMs: number, backoffOnRateLimitMs: number) => {
  const existing = statpalCache.get(url);
  if (existing) {
    if (Date.now() < existing.nextAttemptAt) return existing;
    if (Date.now() - existing.fetchedAt < ttlMs) return existing;
  }

  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), 4500);
  try {
    let fetched: { ok: boolean; status: number; json: unknown; rateLimited: boolean } = { ok: false, status: 0, json: null, rateLimited: false };
    try {
      fetched = await statpalFetchJson(url, controller);
    } catch {
      fetched = { ok: false, status: 0, json: null, rateLimited: false };
    }
    const entry: StatPalCached = {
      ok: fetched.ok,
      status: fetched.status,
      json: fetched.json,
      fetchedAt: Date.now(),
      nextAttemptAt: fetched.rateLimited ? Date.now() + backoffOnRateLimitMs : 0,
      rateLimited: fetched.rateLimited
    };
    statpalCache.set(url, entry);
    return entry;
  } finally {
    clearTimeout(timeoutId);
  }
};

      contentType: fetched.contentType,

const statpalGetRequestCountForKey = async (accessKey: string) => {
  if (!accessKey) return { ok: false, status: 0, requestCount: 0, accessKey: "" };

  const root = getObj(fetched.json);

  const requestCount = Number(typeof requestCountRaw === "string" ? requestCountRaw : requestCountRaw ?? 0);
  return { ok: fetched.ok, status: fetched.status, requestCount: Number.isFinite(requestCount) ? requestCount : 0, accessKey };
};

const statpalPickKeyForBudget = async () => {
  const keys = STATPAL_ACCESS_KEYS.length ? STATPAL_ACCESS_KEYS : STATPAL_ACCESS_KEY ? [STATPAL_ACCESS_KEY] : [];
  if (keys.length === 0) return { key: "", counts: [] as Array<{ keyTail: string; ok: boolean; requestCount: number; status: number }> };
  const countsRaw = await Promise.all(keys.map((k) => statpalGetRequestCountForKey(k)));
  const counts = countsRaw.map((c) => ({ keyTail: c.accessKey.slice(-6), ok: c.ok, requestCount: c.requestCount, status: c.status }));
  const viable = countsRaw.filter((c) => c.ok && c.requestCount < STATPAL_MAX_REQUESTS_PER_DAY);
  const chosen = viable.length ? viable.sort((a, b) => a.requestCount - b.requestCount)[0] : countsRaw[0];
  return { key: chosen?.accessKey ?? "", counts };
};

const fetchStatPal = async (): Promise<{ fixtures: EventFixture[]; rateLimited: boolean }> => {
  if (!STATPAL_ACCESS_KEYS.length) return { fixtures: [], rateLimited: false };

  const out: EventFixture[] = [];
  let rateLimited = false;

  const budget = await statpalPickKeyForBudget();
  const preferKey = budget.key || STATPAL_ACCESS_KEY;
  const allOverLimit = budget.counts.length > 0 && budget.counts.every((c) => c.ok && c.requestCount >= STATPAL_MAX_REQUESTS_PER_DAY);
  if (allOverLimit) return { fixtures: [], rateLimited: true };

  const fetchJson = async (pathname: string, query: Record<string, unknown>, ttlMs: number) => {
    const fetched = await statpalGetCachedTryKeys(pathname, query, ttlMs, 10 * 60_000, preferKey);
    if (fetched.rateLimited) rateLimited = true;
    return fetched;
  };

  const toArr = (x: unknown) => (Array.isArray(x) ? x : x ? [x] : []);
  const asStr = (x: unknown) => String(x ?? "").trim();
  const canon = (s: string) =>
    asStr(s)
      .toLowerCase()
      .replace(/[’'`]/g, "")
      .replace(/[^a-z0-9]+/g, " ")
      .trim();
  const isObj = (x: unknown): x is Record<string, unknown> => Boolean(x) && typeof x === "object" && !Array.isArray(x);

  const pickMajorSoccerLeagueIds = async () => {
    const fetched = await fetchJson("/api/v2/soccer/leagues", {}, 6 * 60 * 60_000);
    if (rateLimited) return [];
    if (!fetched.ok) return [];

    const wanted: Array<{ key: string; match: (league: string, country: string) => boolean }> = [
      { key: "UCL", match: (l) => l.includes("uefa") && l.includes("champions league") },
      { key: "UEL", match: (l) => l.includes("uefa") && l.includes("europa league") && !l.includes("conference") },
      { key: "UECL", match: (l) => l.includes("uefa") && l.includes("europa conference league") },

      { key: "EPL", match: (l, c) => l === "premier league" && c === "england" },
      { key: "EFL", match: (l, c) => l === "championship" && c === "england" },

      { key: "LL", match: (l, c) => (l === "laliga" || l === "la liga") && c === "spain" },
      { key: "SA", match: (l, c) => l === "serie a" && c === "italy" },
      { key: "BL", match: (l, c) => l === "bundesliga" && c === "germany" },
      { key: "L1", match: (l, c) => l === "ligue 1" && c === "france" },

      { key: "LIB", match: (l) => l.includes("libertadores") },
      { key: "SUD", match: (l) => l.includes("sudamericana") }
    ];

    const found: Array<{ id: string; name: string; country: string }> = [];
    const seen = new Set<unknown>();
    const q: unknown[] = [fetched.json];
    while (q.length && found.length < 60) {
      const cur = q.shift();
      if (!cur || seen.has(cur)) continue;
      seen.add(cur);
      if (Array.isArray(cur)) {
        for (const v of cur) q.push(v);
        continue;
      }
      if (!isObj(cur)) continue;
      const id = asStr((cur as any).id ?? (cur as any).league_id ?? (cur as any).leagueId);
      const name = asStr((cur as any).name ?? (cur as any).league_name ?? (cur as any).leagueName ?? (cur as any).league);
      const country = canon(asStr((cur as any).country ?? (cur as any).country_name ?? (cur as any).countryName));
      if (id && name) {
        const l = canon(name);
        if (wanted.some((w) => w.match(l, country))) found.push({ id, name, country });
      }
      for (const v of Object.values(cur)) q.push(v);
    }

    const uniq = new Map<string, string>();
    for (const x of found) uniq.set(x.id, x.name);
    return Array.from(uniq.keys());
  };

    const baseOffset = Number(STATPAL_SOCCER_DAILY_OFFSET) || 0;
    const offsets = [baseOffset, baseOffset + 1];
    const leagueIdsFromDaily = new Set<string>();

    for (const off of offsets) {
      const dailyFetched = await fetchJson("/api/v2/soccer/matches/daily", { offset: String(off) }, 45_000);
      if (rateLimited) return { fixtures: [], rateLimited };
      if (!dailyFetched.ok) continue;
      const root = getObj(dailyFetched.json);
      const leaguesNode =
        (root ? getPath(root, ["live_matches", "league"]) : null) ??
        (root ? getPath(root, ["liveMatches", "league"]) : null) ??
        (root ? getPath(root, ["matches", "league"]) : null) ??
        null;
      const leagues = toArr(leaguesNode);
      for (const l of leagues) {
        const lo = getObj(l);
        if (!lo) continue;
        const leagueId = String((lo as Record<string, unknown>).id ?? "").trim();
        if (leagueId) leagueIdsFromDaily.add(leagueId);
      }
    }

    const majorLeagueIds = await pickMajorSoccerLeagueIds();
    const leagueOrder: string[] = [];
    const seenLeague = new Set<string>();
    const pushLeague = (id: string) => {
      const v = String(id ?? "").trim();
      if (!v || seenLeague.has(v)) return;
      seenLeague.add(v);
      leagueOrder.push(v);
    };
    for (const id of majorLeagueIds) pushLeague(id);
    for (const id of leagueIdsFromDaily) pushLeague(id);

    let soccerCount = 0;
    const maxLeagues = 34;
    for (const leagueId of leagueOrder.slice(0, maxLeagues)) {
      const prematchFetched = await fetchJson(`/api/v2/soccer/leagues/${encodeURIComponent(leagueId)}/odds/prematch`, {}, 60_000);
      if (rateLimited) return { fixtures: [], rateLimited };
      if (!prematchFetched.ok) continue;
      const fixtures = parseSoccerV2PrematchLeagueOdds(prematchFetched.json);
      if (fixtures.length === 0) continue;
      out.push(...fixtures);
      soccerCount += fixtures.length;
      if (soccerCount >= 250) break;
    }

    const nbaFetched = await fetchJson("/api/v1/nba/odds", {}, 30_000);
    if (rateLimited) return { fixtures: [], rateLimited };
    if (nbaFetched.ok) out.push(...parseNbaOddsFixtures(nbaFetched.json));

    const nhlFetched = await fetchJson("/api/v1/nhl/odds", {}, 30_000);
    if (rateLimited) return { fixtures: [], rateLimited };
    if (nhlFetched.ok) out.push(...parseNhlOddsFixtures(nhlFetched.json));

    const mlbFetched = await fetchJson("/api/v1/mlb/odds", {}, 30_000);
    if (rateLimited) return { fixtures: [], rateLimited };
    if (mlbFetched.ok) out.push(...parseMlbOddsFixtures(mlbFetched.json));

    const tennisOddsFetched = await fetchJson("/api/v1/tennis/odds", {}, 25_000);
    const tennisLiveFetched = await fetchJson("/api/v1/tennis/livescores", {}, 10_000);
    if (rateLimited) return { fixtures: [], rateLimited };
    if (tennisOddsFetched.ok) {
      const tennisFixtures = parseTennisOddsFixtures(tennisOddsFetched.json);
      out.push(...(tennisLiveFetched.ok ? applyTennisLivescoresToFixtures(tennisFixtures, tennisLiveFetched.json) : tennisFixtures));
    }

  const dedup = new Map<string, EventFixture>();
  for (const f of out) dedup.set(f.id, f);

  const now = Date.now();
  const minKeep = now - 2 * 60 * 60_000;
  const kept = Array.from(dedup.values()).filter((f) => {
    const t = new Date(f.startsAt).getTime();
    if (Number.isNaN(t)) return false;
    return t >= minKeep;
  });

  return { fixtures: kept, rateLimited };
};

const getFixtures = async () => {
  const ttlMs = 60_000;
  const backoffOnErrorMs = 60_000;
  const backoffOnRateLimitMs = 10 * 60_000;
  if (Date.now() < nextAttemptAt) return cache;
  if (!STATPAL_ACCESS_KEYS.length) return { fixtures: [], fetchedAt: Date.now(), source: "statpal" as const };
  if (Date.now() - cache.fetchedAt < ttlMs) return cache;

  try {
    const fetched = await fetchStatPal();
    if (fetched.fixtures.length > 0) {
      cache = { fixtures: fetched.fixtures, fetchedAt: Date.now(), source: "statpal" };
      nextAttemptAt = Date.now() + ttlMs;
      return cache;
    }
    cache = { fixtures: [], fetchedAt: Date.now(), source: "statpal" };
    nextAttemptAt = Date.now() + (fetched.rateLimited ? backoffOnRateLimitMs : backoffOnErrorMs);
    return cache;
  } catch {
    cache = { fixtures: [], fetchedAt: Date.now(), source: "statpal" };
    nextAttemptAt = Date.now() + backoffOnErrorMs;
    return cache;
  }
};

const findPrice = (fx: EventFixture[], leg: { fixtureId: string; market: MarketKey; selection: SelectionKey; line?: number }) => {
  const f = fx.find((x) => x.id === leg.fixtureId);
  if (!f) return null;
  const m = f.markets.find((x) => x.key === leg.market && (leg.line == null || x.line === leg.line));
  if (m) {
    const o = m.odds.find((x) => x.selection === leg.selection);
    if (!o) return null;
    return { price: o.price, line: m.line };
  }
  return null;
};

const findLivePrice = (leg: { fixtureId: string; market: MarketKey; selection: SelectionKey; line?: number }) => {
  if (!leg.fixtureId.startsWith("Live:")) return null;
  const matchId = leg.fixtureId.slice("Live:".length);
  const match = liveCache.matches.find((m) => m.id === matchId);
  if (!match) return null;
  const market =
    match.markets.find((x) => String(x.name ?? "").toLowerCase().includes("fulltime")) ??
    match.markets.find((x) => String(x.name ?? "").toLowerCase().includes("result")) ??
    match.markets.find((x) => String(x.name ?? "").toLowerCase().includes("1x2")) ??
    match.markets[0];
  const book = (market?.bookmaker ?? [])[0];
  const odds = book?.odd ?? [];

  const minuteRaw = Number(String(match.minute ?? "").replace(/[^\d.]/g, ""));
  const minute = Number.isFinite(minuteRaw) ? minuteRaw : 0;
  const scorePair =
    parseScorePair(`${match.home.score ?? ""}:${match.away.score ?? ""}`) ??
    (typeof match.score === "string" ? parseScorePair(match.score) : null) ??
    null;
  const homeScore = scorePair?.home ?? 0;
  const awayScore = scorePair?.away ?? 0;

  const computeLiveSoccerProbs = (args: { homeScore: number; awayScore: number; minute: number }) => {
    const minute2 = clamp(args.minute, 0, 120);
    const hs0 = Math.max(0, Math.floor(args.homeScore));
    const as0 = Math.max(0, Math.floor(args.awayScore));
    const remaining = Math.max(0, 90 - minute2);
    const frac = remaining / 90;

    const baseHomeFull = 1.35;
    const baseAwayFull = 1.05;
    let lambdaHome = baseHomeFull * frac;
    let lambdaAway = baseAwayFull * frac;

    const diff = hs0 - as0;
    if (diff > 0) {
      lambdaHome *= clamp(1 - 0.08 * diff, 0.25, 1.25);
      lambdaAway *= clamp(1 + 0.12 * diff, 0.25, 2.4);
    } else if (diff < 0) {
      const d = -diff;
      lambdaAway *= clamp(1 - 0.08 * d, 0.25, 1.25);
      lambdaHome *= clamp(1 + 0.12 * d, 0.25, 2.4);
    }

    const maxGoals = 10;
    const pH = poissonPmfUpTo(lambdaHome, maxGoals);
    const pA = poissonPmfUpTo(lambdaAway, maxGoals);
    let pHomeWin = 0;
    let pDraw = 0;
    let pAwayWin = 0;
    for (let i = 0; i <= maxGoals; i += 1) {
      for (let j = 0; j <= maxGoals; j += 1) {
        const p = pH[i] * pA[j];
        const hs = hs0 + i;
        const as = as0 + j;
        if (hs > as) pHomeWin += p;
        else if (hs < as) pAwayWin += p;
        else pDraw += p;
      }
    }
    const s = pHomeWin + pDraw + pAwayWin;
    return {
      lambdaHome,
      lambdaAway,
      pHomeWin: s > 0 ? pHomeWin / s : 1 / 3,
      pDraw: s > 0 ? pDraw / s : 1 / 3,
      pAwayWin: s > 0 ? pAwayWin / s : 1 / 3
    };
  };

  const probs = computeLiveSoccerProbs({ homeScore, awayScore, minute });

  if (leg.market === "1X2") {
    const wanted = leg.selection === "HOME" ? "home" : leg.selection === "DRAW" ? "draw" : "away";
    const found = odds.find((o) => String(o.name ?? "").toLowerCase().includes(wanted));
    const price = Number(found?.value ?? "");
    if (Number.isFinite(price)) return { price, line: undefined as number | undefined };
    const syn = match.synthetic1x2;
    if (!syn) return null;
    const p = leg.selection === "HOME" ? syn.home : leg.selection === "DRAW" ? syn.draw : syn.away;
    return { price: p, line: undefined as number | undefined };
  }

  if (leg.market === "DoubleChance") {
    const p =
      leg.selection === "HOME_DRAW"
        ? probs.pHomeWin + probs.pDraw
        : leg.selection === "HOME_AWAY"
          ? probs.pHomeWin + probs.pAwayWin
          : leg.selection === "DRAW_AWAY"
            ? probs.pDraw + probs.pAwayWin
            : null;
    if (p == null) return null;
    const implied = clamp(p * 1.07, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line: undefined as number | undefined };
  }

  if (leg.market === "DrawNoBet") {
    const denom = probs.pHomeWin + probs.pAwayWin;
    const p = denom > 0 ? (leg.selection === "HOME" ? probs.pHomeWin / denom : leg.selection === "AWAY" ? probs.pAwayWin / denom : null) : null;
    if (p == null) return null;
    const implied = clamp(p * 1.06, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line: undefined as number | undefined };
  }

  if (leg.market === "BTTS") {
    const homeScored = homeScore > 0;
    const awayScored = awayScore > 0;
    const pHomeAtLeastOne = homeScored ? 1 : 1 - Math.exp(-probs.lambdaHome);
    const pAwayAtLeastOne = awayScored ? 1 : 1 - Math.exp(-probs.lambdaAway);
    const pYes = clamp(pHomeAtLeastOne, 0, 1) * clamp(pAwayAtLeastOne, 0, 1);
    const p = leg.selection === "YES" ? pYes : leg.selection === "NO" ? 1 - pYes : null;
    if (p == null) return null;
    const implied = clamp(clamp(p, 1e-9, 0.999999) * 1.06, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line: undefined as number | undefined };
  }

  if (leg.market === "Handicap") {
    const line = typeof leg.line === "number" ? leg.line : null;
    if (line == null) return null;
    if (leg.selection !== "HOME" && leg.selection !== "AWAY") return null;

    const maxGoals = 10;
    const pH = poissonPmfUpTo(probs.lambdaHome, maxGoals);
    const pA = poissonPmfUpTo(probs.lambdaAway, maxGoals);
    let pHomeCover = 0;
    let pAwayCover = 0;
    let pPush = 0;
    for (let i = 0; i <= maxGoals; i += 1) {
      for (let j = 0; j <= maxGoals; j += 1) {
        const p = pH[i] * pA[j];
        const v = (homeScore + i) + line - (awayScore + j);
        if (v > 0) pHomeCover += p;
        else if (v < 0) pAwayCover += p;
        else pPush += p;
      }
    }
    const denom = clamp(1 - pPush, 1e-9, 1);
    const p = leg.selection === "HOME" ? pHomeCover / denom : pAwayCover / denom;
    const implied = clamp(clamp(p, 1e-9, 0.999999) * 1.06, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line };
  }

  if (leg.market === "OverUnder") {
    const line = typeof leg.line === "number" ? leg.line : null;
    if (line == null) return null;
    const currentTotal = homeScore + awayScore;
    const neededUnderK = Math.floor(line - currentTotal);
    const under = neededUnderK < 0 ? 0 : poissonCdf(probs.lambdaHome + probs.lambdaAway, neededUnderK);
    const over = 1 - under;
    const p = leg.selection === "OVER" ? over : leg.selection === "UNDER" ? under : null;
    if (p == null) return null;
    const implied = clamp(clamp(p, 1e-9, 0.999999) * 1.06, 1e-9, 0.999999);
    return { price: round2(clamp(1 / implied, 1.01, 100)), line };
  }

  if (leg.market === "CorrectScore") {
    const price = computeSoccerCorrectScoreTopPrice({
      lambdaHome: probs.lambdaHome,
      lambdaAway: probs.lambdaAway,
      homeScore,
      awayScore,
      selection: String(leg.selection),
      overround: 1.08
    });
    return price == null ? null : { price, line: undefined as number | undefined };
  }

  return null;
};

app.get("/odds/statpal/mlb/odds/slim", requireStatpalKey, async (_req, res) => {
  const fetched = await statpalGetCachedTryKeys("/api/v1/mlb/odds", {}, 5_000, 15_000);
  if (!fetched.ok) return res.status(fetched.status || 502).json(fetched.json ?? { error: "STATPAL_ERROR" });

  const root = getObj(fetched.json);
  const category = root ? getObj(getPath(root, ["odds", "category"])) : null;
  const matchArr = category ? getPath(category, ["matches", "match"]) : null;
  const matches = Array.isArray(matchArr) ? (matchArr as unknown[]) : [];

  const pickBookmaker = (bookmakers: unknown[]) => {
    const score = (b: unknown) => {
      const bo = getObj(b);
      if (!bo) return -1;
      const oddArr = Array.isArray((bo as Record<string, unknown>).odd) ? ((bo as Record<string, unknown>).odd as unknown[]) : [];
      return oddArr.length;
    };
    const sorted = [...bookmakers].sort((a, b) => score(b) - score(a));
    return getObj(sorted[0]);
  };

  const selectionFromOddName = (nameRaw: string): SelectionKey | null => {
    const n = canon(nameRaw);
    if (!n) return null;
    if (n.includes("home") || n === "1") return "HOME";
    if (n.includes("draw") || n === "x") return "DRAW";
    if (n.includes("away") || n === "2") return "AWAY";
    return null;
  };

  const out: Array<{ id: string; odds1x2?: { home?: number; draw?: number; away?: number } }> = [];
  for (const m of matches) {
    const mo = getObj(m);
    if (!mo) continue;
    const id = String((mo as Record<string, unknown>).mlbid ?? (mo as Record<string, unknown>).id ?? "").trim();
    if (!id) continue;

    const typeArr = getPath(mo, ["odds", "type"]);
    const types = Array.isArray(typeArr) ? (typeArr as unknown[]) : [];
    let best: { home?: number; draw?: number; away?: number } | null = null;

    for (const t of types) {
      const to = getObj(t);
      if (!to) continue;
      const v = canon(String((to as Record<string, unknown>).value ?? ""));
      if (!v) continue;
      if (!(v.includes("3way result") || v.includes("moneyline") || v.includes("home/away") || v.includes("match winner") || v.includes("winner") || v.includes("result")))
        continue;
      const booksNode = getPath(to, ["bookmaker"]);
      const books = Array.isArray(booksNode) ? (booksNode as unknown[]) : booksNode ? [booksNode] : [];
      const book = pickBookmaker(books);
      if (!book) continue;
      const oddArr = (book as Record<string, unknown>).odd;
      const odds = Array.isArray(oddArr) ? (oddArr as unknown[]) : oddArr ? [oddArr] : [];
      const got: { home?: number; draw?: number; away?: number } = {};
      for (const o of odds) {
        const oo = getObj(o);
        if (!oo) continue;
        const sel = selectionFromOddName(String((oo as Record<string, unknown>).name ?? ""));
        const priceRaw = normalizeOdd((oo as Record<string, unknown>).value);
        if (!sel || priceRaw == null || !Number.isFinite(priceRaw) || priceRaw <= 1.0 || priceRaw >= 200) continue;
        if (sel === "HOME") got.home = round2(priceRaw);
        if (sel === "DRAW") got.draw = round2(priceRaw);
        if (sel === "AWAY") got.away = round2(priceRaw);
      }
      if (got.home != null && got.away != null) {
        best = got;
        break;
      }
    }

    out.push({ id, odds1x2: best ?? undefined });
  }

  return res.json({ ok: true, fetchedAt: Date.now(), matches: out });
});

  const selectionFromOddName = (nameRaw: string): SelectionKey | null => {
    const n = canon(nameRaw);
    if (!n) return null;
    if (n.includes("home") || n === "1") return "HOME";
    if (n.includes("draw") || n === "x") return "DRAW";
    if (n.includes("away") || n === "2") return "AWAY";
    return null;
  };

  const out: Array<{ id: string; fix_id?: string; odds1x2?: { home?: number; draw?: number; away?: number } }> = [];
  for (const m of matches) {
    const mo = getObj(m);
    if (!mo) continue;
    const id = String((mo as Record<string, unknown>).id ?? "").trim();
    const fixId = String((mo as Record<string, unknown>).fix_id ?? (mo as Record<string, unknown>).fixId ?? "").trim();
    if (!id) continue;

    const typeArr = getPath(mo, ["odds", "type"]);
    const types = Array.isArray(typeArr) ? (typeArr as unknown[]) : [];
    let best: { home?: number; draw?: number; away?: number } | null = null;

    for (const t of types) {
      const to = getObj(t);
      if (!to) continue;
      const v = canon(String((to as Record<string, unknown>).value ?? ""));
      if (!v) continue;
      if (!(v.includes("3way result") || v.includes("1x2") || v.includes("full time result") || v.includes("moneyline") || v.includes("home/away"))) continue;
      const booksNode = getPath(to, ["bookmaker"]);
      const books = Array.isArray(booksNode) ? (booksNode as unknown[]) : booksNode ? [booksNode] : [];
      const book = pickBookmaker(books);
      if (!book) continue;
      const oddArr = (book as Record<string, unknown>).odd;
      const odds = Array.isArray(oddArr) ? (oddArr as unknown[]) : oddArr ? [oddArr] : [];
      const got: { home?: number; draw?: number; away?: number } = {};
      for (const o of odds) {
        const oo = getObj(o);
        if (!oo) continue;
        const sel = selectionFromOddName(String((oo as Record<string, unknown>).name ?? ""));
        const priceRaw = normalizeOdd((oo as Record<string, unknown>).value);
        if (!sel || priceRaw == null || !Number.isFinite(priceRaw) || priceRaw <= 1.0 || priceRaw >= 200) continue;
        if (sel === "HOME") got.home = round2(priceRaw);
        if (sel === "DRAW") got.draw = round2(priceRaw);
        if (sel === "AWAY") got.away = round2(priceRaw);
      }
      if (got.home != null && got.away != null) {
        best = got;
        break;
      }
    }

    out.push({ id, fix_id: fixId || undefined, odds1x2: best ?? undefined });
  }

  return res.json({ ok: true, fetchedAt: Date.now(), matches: out });
});

const parseScorePair = (s: string) => {
  const m = /(\d+)\D+(\d+)/.exec(s);
  if (!m) return null;
  const a = Number(m[1]);
  const b = Number(m[2]);
  if (!Number.isFinite(a) || !Number.isFinite(b)) return null;
  return { home: a, away: b };
};

const computeLive1x2Model = (args: { homeScore: number; awayScore: number; minute: number }) => {
  const minute = clamp(args.minute, 0, 120);
  const homeScore = Math.max(0, Math.floor(args.homeScore));
  const awayScore = Math.max(0, Math.floor(args.awayScore));
  const remaining = Math.max(0, 90 - minute);
  const frac = remaining / 90;

  const baseHomeFull = 1.35;
  const baseAwayFull = 1.05;
  let lambdaHome = baseHomeFull * frac;
  let lambdaAway = baseAwayFull * frac;

  const diff = homeScore - awayScore;
  if (diff > 0) {
    lambdaHome *= clamp(1 - 0.08 * diff, 0.25, 1.25);
    lambdaAway *= clamp(1 + 0.12 * diff, 0.25, 2.4);
  } else if (diff < 0) {
    const d = -diff;
    lambdaAway *= clamp(1 - 0.08 * d, 0.25, 1.25);
    lambdaHome *= clamp(1 + 0.12 * d, 0.25, 2.4);
  }

  const maxGoals = 10;
  const pH = poissonPmfUpTo(lambdaHome, maxGoals);
  const pA = poissonPmfUpTo(lambdaAway, maxGoals);
  let pHomeWin = 0;
  let pDraw = 0;
  let pAwayWin = 0;
  for (let i = 0; i <= maxGoals; i += 1) {
    for (let j = 0; j <= maxGoals; j += 1) {
      const p = pH[i] * pA[j];
      const hs = homeScore + i;
      const as = awayScore + j;
      if (hs > as) pHomeWin += p;
      else if (hs < as) pAwayWin += p;
      else pDraw += p;
    }
  }
  const s = pHomeWin + pDraw + pAwayWin;
  const p1 = s > 0 ? pHomeWin / s : 1 / 3;
  const px = s > 0 ? pDraw / s : 1 / 3;
  const p2 = s > 0 ? pAwayWin / s : 1 / 3;

  const [home, draw, away] = probsToOdds([clamp(p1, 1e-6, 0.999), clamp(px, 1e-6, 0.999), clamp(p2, 1e-6, 0.999)], 1.07);
  return { home, draw, away };
};

let liveCache: LiveOddsPayload = { fetchedAt: 0, matches: [] };
let liveNextAttemptAt = 0;
let livePolling = false;

const getLiveOdds = async (): Promise<LiveOddsPayload> => {
  if (!STATPAL_ACCESS_KEYS.length) return { fetchedAt: Date.now(), matches: [] };
  if (Date.now() < liveNextAttemptAt) return liveCache;
  if (Date.now() - liveCache.fetchedAt < 8_000) return liveCache;

  const budget = await statpalPickKeyForBudget();
  const allOverLimit = budget.counts.length > 0 && budget.counts.every((c) => c.ok && c.requestCount >= STATPAL_MAX_REQUESTS_PER_DAY);
  if (allOverLimit) {
    liveNextAttemptAt = Date.now() + 10 * 60_000;
    return liveCache;
  }

  const fetched = await statpalGetCachedTryKeys(
    "/api/v2/soccer/odds/live",
    STATPAL_LIVE_ODDS_MARKET_ID ? { market_id: STATPAL_LIVE_ODDS_MARKET_ID } : {},
    8_000,
    60_000,
    budget.key || STATPAL_ACCESS_KEY
  );
  if (!fetched.ok) {
    liveNextAttemptAt = Date.now() + (fetched.status === 429 ? 10 * 60_000 : 60_000);
    return liveCache;
  }
  const parsed = parseLiveOddsPayload(fetched.json);
  const marketIndex = await getSoccerLiveMarketsIndex();
  if (marketIndex.size > 0) {
    for (const match of parsed.matches) {
      for (const market of match.markets ?? []) {
        if (market.name && String(market.name).trim()) continue;
        const id = String(market.id ?? "").trim();
        if (!id) continue;
        const resolved = marketIndex.get(id);
        if (resolved) market.name = resolved;
      }
    }
  }
  liveCache = parsed;
  liveNextAttemptAt = Date.now() + 8_000;
  return liveCache;
};

app.get("/odds/live", async (_req, res) => {
  const data = await getLiveOdds();
  return res.json(data);
});

app.get("/odds/statpal/soccer/images", requireStatpalKey, async (req, res) => {
  const url = buildStatPalUrl("/api/v2/soccer/images", req.query as Record<string, unknown>);
  const fetched = await statpalGetBinaryCached(url, "image/png, application/json", 24 * 60 * 60_000, 10 * 60_000);
  if (!fetched.ok) {
    if (fetched.contentType.includes("application/json")) {
      try {
        const json = JSON.parse(fetched.body.toString("utf8")) as unknown;
        return res.status(fetched.status || 502).json(json);
      } catch {
        return res.status(fetched.status || 502).json({ error: "STATPAL_ERROR" });
      }
    }
    return res.status(fetched.status || 502).json({ error: "STATPAL_ERROR" });
  }
  if (fetched.contentType) res.setHeader("content-type", fetched.contentType);
  return res.send(fetched.body);
});

app.get("/odds/provider-status", async (_req, res) => {
  const keyPresent = STATPAL_ACCESS_KEYS.length > 0;
  if (!keyPresent) {
    return res.json({
      provider: "statpal",
      keyPresent,
      keyLength: 0,
      requestCount: 0,
      maxRequestsPerDay: STATPAL_MAX_REQUESTS_PER_DAY,
      overLimit: false
    });
  }

  const budget = await statpalPickKeyForBudget();
  const okCounts = budget.counts.filter((c) => c.ok).map((c) => c.requestCount);
  const requestCount = okCounts.length ? Math.min(...okCounts) : 0;
  const overLimit = budget.counts.length > 0 && budget.counts.every((c) => c.ok && c.requestCount >= STATPAL_MAX_REQUESTS_PER_DAY);
  let envFileKeyTail = "";
  let envPath = "";
  let envExists = false;
  try {
    envPath = path.join(repoRoot, "env", ".env");
    envExists = fs.existsSync(envPath);
    if (envExists) {
      const content = fs.readFileSync(envPath, "utf8");
      for (const line of content.split(/\r?\n/)) {
        const trimmed = line.trim();
        if (!trimmed || trimmed.startsWith("#")) continue;
        const eq = trimmed.indexOf("=");
        if (eq <= 0) continue;
        const k = trimmed
          .slice(0, eq)
          .trim()
          .replace(/^\uFEFF/, "");
        if (k !== "STATPAL_ACCESS_KEY") continue;
        const v = trimmed.slice(eq + 1).trim();
        envFileKeyTail = v.slice(-6);
        break;
      }
    }
  } catch {
    envFileKeyTail = "";
  }
  return res.json({
    provider: "statpal",
    keyPresent,
    keyLength: STATPAL_ACCESS_KEY.length,
    keyTail: STATPAL_ACCESS_KEY.slice(-6),
    repoRoot,
    envPath,
    envExists,
    envFileKeyTail,
    requestCount,
    maxRequestsPerDay: STATPAL_MAX_REQUESTS_PER_DAY,
    overLimit,
    keysConfigured: STATPAL_ACCESS_KEYS.map((k) => k.slice(-6)),
    budgetCounts: budget.counts
  });
});

app.post("/odds/validate", async (req, res) => {
  const legs = (req.body?.legs ?? []) as BetLeg[];
  if (!Array.isArray(legs) || legs.length === 0) return res.status(400).json({ error: "INVALID_LEGS" });

  const current = await getFixtures();
  const baseFixtures = current.fixtures;
  const validated = legs.map((leg) => {
    const found = findLivePrice(leg) ?? findPrice(baseFixtures, leg);
    return {
      fixtureId: leg.fixtureId,
      market: leg.market,
      selection: leg.selection,
      requestedPrice: leg.price,
      currentPrice: found?.price ?? null,
      ok: found != null && Math.abs(found.price - leg.price) <= 0.05
    };
  });

  const ok = validated.every((x) => x.ok);
  return res.json({ ok, validated, requestId: nanoid() });
});

const server = http.createServer(app);

const wss = new WebSocketServer({ server, path: "/odds/live/ws" });
const wsClients = new Set<WebSocket>();

const broadcast = (msg: unknown) => {
  const raw = JSON.stringify(msg);
  for (const c of wsClients) {
    if (c.readyState === c.OPEN) c.send(raw);
  }
};

const ensureLivePoller = () => {
  if (livePolling) return;
  livePolling = true;
  const loop = async () => {
    while (wsClients.size > 0) {
      const payload = await getLiveOdds();
      broadcast({ type: "snapshot", serverTime: Date.now(), ...payload });
      await new Promise((r) => setTimeout(r, 10_000));
    }
    livePolling = false;
  };
  loop().catch(() => {
    livePolling = false;
  });
};

wss.on("connection", (socket: WebSocket) => {
  wsClients.add(socket);
  getLiveOdds()
    .then((payload) => socket.send(JSON.stringify({ type: "snapshot", serverTime: Date.now(), ...payload })))
    .catch(() => {});
  ensureLivePoller();
  socket.on("close", () => {
    wsClients.delete(socket);
  });
});

server.listen(PORT, () => {
  process.stdout.write(`odds listening on :${PORT}\n`);
});
