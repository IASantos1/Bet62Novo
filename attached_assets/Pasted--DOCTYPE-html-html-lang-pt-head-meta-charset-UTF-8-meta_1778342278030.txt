    <script>
        let currentBets = [];

        const matchesData = [
            { id: 1, home: "Flamengo", away: "Palmeiras", league: "Brasileirão Série A", time: "19:30", odds: { home: 2.15, draw: 3.40, away: 3.25 } },
            { id: 2, home: "Real Madrid", away: "Barcelona", league: "La Liga", time: "22:00", odds: { home: 2.30, draw: 3.50, away: 3.10 } },
            { id: 3, home: "Manchester City", away: "Liverpool", league: "Premier League", time: "17:00", odds: { home: 1.95, draw: 3.60, away: 3.80 } },
        ];

        function renderMatches() {
            const container = document.getElementById('matches-grid');
            container.innerHTML = '';

            matchesData.forEach(match => {
                const card = document.createElement('div');
                card.className = "match-card bg-zinc-900 border border-zinc-800 rounded-3xl p-6 cursor-pointer";
                
                card.innerHTML = `
                    <div class="flex justify-between text-xs text-zinc-400 mb-4">
                        <span>${match.league}</span>
                        <span>${match.time}</span>
                    </div>
                    <div class="flex justify-between items-center font-semibold text-lg mb-6">
                        <div>${match.home}</div>
                        <div class="text-red-500 text-xl">VS</div>
                        <div class="text-right">${match.away}</div>
                    </div>
                    
                    <div class="grid grid-cols-3 gap-3 text-center">
                        <div onclick="addBet('${match.home}', ${match.odds.home}); event.stopImmediatePropagation()" 
                             class="odds-btn bg-zinc-800 p-4 rounded-2xl cursor-pointer">
                            <div class="text-xs text-zinc-400">Casa</div>
                            <div class="font-bold text-xl">${match.odds.home}</div>
                        </div>
                        <div onclick="addBet('Empate', ${match.odds.draw}); event.stopImmediatePropagation()" 
                             class="odds-btn bg-zinc-800 p-4 rounded-2xl cursor-pointer">
                            <div class="text-xs text-zinc-400">Empate</div>
                            <div class="font-bold text-xl">${match.odds.draw}</div>
                        </div>
                        <div onclick="addBet('${match.away}', ${match.odds.away}); event.stopImmediatePropagation()" 
                             class="odds-btn bg-zinc-800 p-4 rounded-2xl cursor-pointer">
                            <div class="text-xs text-zinc-400">Fora</div>
                            <div class="font-bold text-xl">${match.odds.away}</div>
                        </div>
                    </div>
                `;

                // Clique em qualquer lugar do cartão (exceto odds) abre o modal
                card.addEventListener('click', (e) => {
                    if (!e.target.closest('.odds-btn')) {
                        openMatchModal(match);
                    }
                });

                container.appendChild(card);
            });
        }

        function openMatchModal(match) {
            document.getElementById('modal-league').textContent = `${match.league} • ${match.time}`;
            document.getElementById('modal-teams').innerHTML = `${match.home} <span class="text-red-500">VS</span> ${match.away}`;

            const marketsHTML = `
                <div class="space-y-8">
                    <div>
                        <h4 class="font-bold text-lg mb-4">Mais Mercados</h4>
                        <div class="grid grid-cols-2 gap-4">
                            <div onclick="addBet('Over 2.5', 1.85)" class="odds-btn bg-zinc-800 p-5 rounded-2xl cursor-pointer text-center">Over 2.5<br><span class="font-bold">1.85</span></div>
                            <div onclick="addBet('Under 2.5', 1.95)" class="odds-btn bg-zinc-800 p-5 rounded-2xl cursor-pointer text-center">Under 2.5<br><span class="font-bold">1.95</span></div>
                            <div onclick="addBet('Ambas Marcam - Sim', 1.90)" class="odds-btn bg-zinc-800 p-5 rounded-2xl cursor-pointer text-center">Ambas Marcam<br><span class="font-bold">1.90</span></div>
                        </div>
                    </div>
                </div>
            `;

            document.getElementById('modal-markets').innerHTML = marketsHTML;
            document.getElementById('match-modal').classList.remove('hidden');
        }

        function closeModal() {
            document.getElementById('match-modal').classList.add('hidden');
        }

        function addBet(selection, odd) {
            currentBets.push({ selection, odd });
            renderBetSlip();
            if (window.innerWidth < 1024) toggleMobileSlip(true);
        }

        function renderBetSlip() {
            const html = currentBets.map((bet, i) => `
                <div class="bg-zinc-950 rounded-2xl p-4 mb-3 flex justify-between items-center">
                    <div>
                        <div class="font-medium">${bet.selection}</div>
                        <div class="text-red-400 font-bold">${bet.odd}</div>
                    </div>
                    <button onclick="removeBet(${i}); event.stopImmediatePropagation()" class="text-red-400"><i class="fas fa-trash"></i></button>
                </div>
            `).join('');

            document.getElementById('bet-slip-desktop').innerHTML = html || `<div class="text-center py-20 text-zinc-400">Boletim vazio</div>`;
            document.getElementById('bet-slip-mobile-content').innerHTML = html || `<div class="text-center py-20 text-zinc-400">Boletim vazio</div>`;
        }

        function removeBet(i) {
            currentBets.splice(i, 1);
            renderBetSlip();
        }

        function toggleMobileSlip(open = null) {
            const slip = document.getElementById('bet-slip-mobile');
            open === null ? slip.classList.toggle('open') : open ? slip.classList.add('open') : slip.classList.remove('open');
        }

        function toggleSidebar() {
            document.getElementById('sidebar').classList.toggle('open');
        }

        function showSection(section) {
            document.getElementById('sports-section').classList.toggle('hidden', section !== 'sports');
            document.getElementById('promotions-section').classList.toggle('hidden', section !== 'promotions');
        }

        function showAuthModal() {
            alert("🔐 Login / Registro");
        }

        function claimBonus() {
            alert("🎉 Bônus ativado com sucesso!");
        }

        window.onload = renderMatches;
    </script>
