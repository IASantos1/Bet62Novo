⚙️ ENGINE DE SUSPENSÃO
function suspendGoalMarkets(match) {

  Object.entries(MARKET_CONFIG)
    .forEach(([market, config]) => {

      if (!config.suspend) return;

      match.markets[market].status =
        "suspended";

      setTimeout(() => {

        recalculateMarket(match, market);

        match.markets[market].status =
          "open";

      }, config.delay);

    });
}
